3   import info.smart_tools.smartactors.core.db_storage.exceptions.QueryBuildException;
4   
5   import java.util.regex.Pattern;
6   
7   /**
8    * Wrapper for collection name string
9    */
10  public class CollectionName {
11  
12      private static Pattern VALIDATION_PATTERN = Pattern.compile("[a-zA-Z_][0-9a-zA-Z_]*");
13  
14      private String name;
15  
16      private CollectionName(String name) {
17          this.name = name;
18      }
19  
20      public String toString() {
21          return this.name;
22      }
23  
24      /**
25       * Create collection name from input string
26       * @param name string with name
27       * @return created CollectionName
28       * @throws QueryBuildException if input string doesn't match validation pattern
29       */
30      public static CollectionName fromString(String name) throws QueryBuildException {
31  
32          if (!VALIDATION_PATTERN.matcher(name).matches()) {
33              throw new QueryBuildException("Invalid collection name: "+name);
34          }
35  
36          return new CollectionName(name.toLowerCase());
37      }
38  }
