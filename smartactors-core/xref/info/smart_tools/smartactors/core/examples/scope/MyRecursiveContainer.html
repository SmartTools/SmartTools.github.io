3   import java.util.Map;
4   import java.util.concurrent.ConcurrentHashMap;
5   
6   /**
7    *  A sample of container which takes care on the parent-child relationships.
8    *  It's {@link #get} method queries parent container if cannot find the requested value in it's own storage.
9    */
10  public class MyRecursiveContainer<K, V> {
11  
12      private final MyRecursiveContainer<K, V> parent;
13  
14      private final Map<K, V> storage = new ConcurrentHashMap<>();
15  
16      public MyRecursiveContainer(final MyRecursiveContainer<K, V> parent) {
17          this.parent = parent;
18      }
19  
20      public V get(final K key) {
21          V result = null;
22          result = storage.get(key);
23          if (result == null && parent != null) {
24              result = parent.get(key);
25          }
26          return result;
27      }
28  
29      public void put(final K key, final V value) {
30          storage.put(key, value);
31      }
32  
33  }
