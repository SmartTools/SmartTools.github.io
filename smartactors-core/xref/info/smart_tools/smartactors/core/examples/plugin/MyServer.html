20   *  Sample server implementation.

22  public class MyServer implements IServer {
23  
24      @Override
25      public void initialize() throws ServerInitializeException {
26          try {
27              IBootstrap bootstrap = new Bootstrap();         // initializes the server in correct order
28              IPluginCreator creator = new PluginCreator();   // instantiates the plugin correctly
29              IPluginLoaderVisitor<String> visitor = new MyPluginVisitor();   // checks the plugin loadings
30  
31              ClassLoader urlClassLoader =
32                      new ExpansibleURLClassLoader(new URL[]{}, ClassLoader.getSystemClassLoader());  // loads plugins' classes
33  
34              IPluginLoader<String> pluginLoader = new PluginLoader(          // loads plugins to classloader
35                      urlClassLoader,
36                      (t) -> {
37                          try {
38                              IPlugin plugin = creator.create(t, bootstrap);  // creates the plugin instance
39                              plugin.load();                                  // loads the plugin
40                          } catch (Exception e) {
41                              throw new RuntimeException("Could not create instance of IPlugin", e);
42                          }
43                      },
44                      visitor
45              );
46  
47              File examplesJar = new File(System.getProperty("user.home"), ".m2/repository/info/smart_tools/smartactors/" +
48                      "core.examples/1.0-SNAPSHOT/core.examples-1.0-SNAPSHOT.jar");
49              pluginLoader.loadPlugin(examplesJar.getAbsolutePath());     // loads plugins
50              bootstrap.start();                                          // starts initialization
51          } catch (Throwable e) {
52              throw new ServerInitializeException("Server initialization failed.", e);
53          }
54      }
55  
56      @Override
57      public void start() throws ServerExecutionException {
58      }
59  }
