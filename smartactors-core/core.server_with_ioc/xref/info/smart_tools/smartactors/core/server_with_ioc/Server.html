24      /**
25       * Initialize service locators
26       * @throws ServerInitializeException if any errors occurred
27       */
28      public void initialize()
29              throws ServerInitializeException {
30          try {
31              initializeScopeProvider();
32          } catch (Exception e) {
33              throw new ServerInitializeException("Server Initialize failed.");
34          }
35      }
36  
37      public void start()
38              throws ServerExecutionException {
39          try {
40  
41              /** Example of registration the Singleton strategy */
42              Key<String> key1 = new Key<String>(String.class, "a");
43              IResolveDependencyStrategy strategy = new SingletonStrategy("abcd");
44              IOC.register(key1, strategy);
45              /** Example of usage the Singleton strategy */
46              String result1 = IOC.resolve(key1);
47  
48              /** Example of registration the ResolveByNameIoc strategy */
49              IOC.register(IOC.getKeyForKeyStorage(), new ResolveByNameIocStrategy(
50                      (a) -> new Key<IKey>((String) a[0]))
51              );
52              /** Example of usage the ResolveByNameIoc strategy */
53              IKey key2 = IOC.resolve(IOC.getKeyForKeyStorage(), "key2");
54              IKey key3 = IOC.resolve(IOC.getKeyForKeyStorage(), "key3");
55              IKey key4 = IOC.resolve(IOC.getKeyForKeyStorage(), "key2");  // Should be same as key2
56  
57              /** Example of registration the CreateNewInstance strategy */
58              IKey<IObject> key5 = new Key<IObject>("create_new");
59              IOC.register(key5, new CreateNewInstanceStrategy(
60                              (a) -> new IObjectImpl())
61              );
62              /** Example of usage the CreateNewInstance strategy */
63              IObject test = IOC.resolve(key5);
64  
65  
66              /** Example of registration and usage CreateNewInstance strategy by named key */
67              Object param = new Object();
68              IKey key6 = IOC.resolve(IOC.getKeyForKeyStorage(), "key6");
69              IOC.register(key6, new CreateNewInstanceStrategy(
70                              (a) -> new IObjectImpl())
71              );
72              IObject obj = IOC.resolve(IOC.resolve(IOC.getKeyForKeyStorage(), "key6"), param);
73              System.out.print("Did it.");
74  
75          } catch (Exception | Error e) {
76              System.out.println(e);
77          }
78      }

80      private void initializeScopeProvider()
81              throws Exception {
82          // /** Subscribe for java 1.7 or less */
83          // ScopeProvider.subscribeOnCreationNewScope(new ScopeCreationEventHandler(IOC.getIocKey()));
84  
85          /** Subscribe for java 1.8 or high */
86          ScopeProvider.subscribeOnCreationNewScope(
87                  scope -> {
88                      try {
89                          scope.setValue(IOC.getIocKey(), new StrategyContainer());
90                      } catch (Exception e) {
91                          throw new Error(e);
92                      }
93                  }
94          );
95  
96  
97          Object keyOfMainScope = ScopeProvider.createScope(null);
98          IScope mainScope = ScopeProvider.getScope(keyOfMainScope);
99          ScopeProvider.setCurrentScope(mainScope);
100     }
