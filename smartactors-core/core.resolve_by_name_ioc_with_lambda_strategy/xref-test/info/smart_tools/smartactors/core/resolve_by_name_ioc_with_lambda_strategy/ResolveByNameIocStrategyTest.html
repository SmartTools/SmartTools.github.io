41      @Test
42      public void checkStrategyResolutionOnceCallInternalStrategy()
43              throws Exception {
44          Counter counter = new Counter();
45          Object o = new Object();
46          IResolveDependencyStrategy strategy = new ResolveByNameIocStrategy(
47                  (args)-> {
48                      ++counter.times;
49                      return o;
50                  }
51          );
52          assertEquals(o, strategy.resolve("key"));
53          assertEquals(o, strategy.resolve("key"));
54          assertEquals(o, strategy.resolve("key"));
