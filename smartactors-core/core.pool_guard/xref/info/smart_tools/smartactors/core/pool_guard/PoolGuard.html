1   package info.smart_tools.smartactors.core.pool_guard;

3   import info.smart_tools.smartactors.core.ipool.IPool;
4   import info.smart_tools.smartactors.core.pool_guard.exception.PoolGuardException;
5   
6   /**
7    * Implementation of {@link IPoolGuard}
8    */
9   public class PoolGuard implements IPoolGuard {
10      private IPool pool;
11      private Object currentObject;
12  
13      /**
14       * Constructor for PoolGuard
15       * @param pool the using pool
16       * @throws PoolGuardException if any errors occurred
17       */
18      public PoolGuard(final IPool pool) throws PoolGuardException {
19          try {
20              this.pool = pool;
21              this.currentObject = pool.take();
22          } catch (Exception e) {
23              throw new PoolGuardException("PoolGuard cannot be initialized", e);
24          }
25      }
26  
27  
28      /**
29       * Get the free item from pool
30       */
31      public Object getObject() {
32          return currentObject;
33      }
34  
35      /**
36       * Return object to pool if object is no needed more
37       */
38      public void close() {
39          try {
40              pool.put(currentObject);
41          } catch (Exception e) {
42              throw new RuntimeException("PoolGuard could not restore current item.", e);
43          }
44      }
45  }
