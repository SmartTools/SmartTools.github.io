33  public class GeneralSQLOrderWriterTest {
34      private ISearchQueryWriter orderWriter;
35  
36      @Before
37      public void setUp() {
38          mockStatic(IOC.class);
39          mockStatic(Keys.class);
40  
41          orderWriter = GeneralSQLOrderWriter.create();
42      }
43  
44      @Test
45      public void should_WritesORDERClauseIntoQueryStatement() throws Exception {
46          ISearchQuery ISearchQuery = mock(ISearchQuery.class);
47          IObject orderItem = mock(IObject.class);
48  
49          FieldPath fieldPath = mock(FieldPath.class);
50          String sortDirection = "testSQLStrDirection";
51  
52          when(ISearchQuery.countOrderBy()).thenReturn(1);
53          when(ISearchQuery.getOrderBy(0)).thenReturn(orderItem);
54          when(orderItem.getValue(anyObject())).thenReturn("testOrderField").thenReturn("testOrderDirection");
55  
56          when(fieldPath.getSQLRepresentation()).thenReturn("testSQLStrField");
57  
58          IKey fieldPathKey = mock(IKey.class);
59          IKey strKey = mock(IKey.class);
60  
61          when(Keys.getOrAdd(PSQLFieldPath.class.toString())).thenReturn(fieldPathKey);
62          when(Keys.getOrAdd(String.class.toString())).thenReturn(strKey);
63  
64          when(IOC.resolve(eq(fieldPathKey), anyString())).thenReturn(fieldPath).thenReturn(sortDirection);
65  
66          QueryStatement queryStatement = new QueryStatement();
67          List<SQLQueryParameterSetter> setters = new LinkedList<>();
68          orderWriter.write(queryStatement, ISearchQuery, setters);
69  
70          assertTrue("ORDER BY(testSQLStrField)ASC,(1)".equals(queryStatement.getBodyWriter().toString()));
71  
72          verify(orderItem, times(2)).getValue(anyObject());
73          verify(fieldPath, times(1)).getSQLRepresentation();
74          verifyStatic(times(1));
75          Keys.getOrAdd(PSQLFieldPath.class.toString());
76          verifyStatic(times(1));
77          Keys.getOrAdd(String.class.toString());
78          verifyStatic(times(1));
79          IOC.resolve(eq(fieldPathKey), anyObject());
80          verifyStatic(times(1));
81          IOC.resolve(eq(strKey), anyObject());
82      }
