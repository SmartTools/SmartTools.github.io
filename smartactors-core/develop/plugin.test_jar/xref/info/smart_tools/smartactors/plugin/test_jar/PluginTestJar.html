9   /**
10   * Test Plugin.
11   * Needed for run tests for IPluginLoader
12   */
13  public class PluginTestJar implements IPlugin {
14  
15      /**
16       * Inner container for incoming instance of {@link IBootstrap}
17       */
18      private IBootstrap<IBootstrapItem<String>> bootstrap;
19  
20      /**
21       * Constructor.
22       * It will be finding by PluginLoader
23       * @param bootstrap instance of {@link IBootstrap}
24       */
25      public PluginTestJar(final IBootstrap<IBootstrapItem<String>> bootstrap) {
26          this.bootstrap = bootstrap;
27      }
28  
29      @Override
30      public void load()
31              throws PluginException {
32          try {
33              IBootstrapItem<String> item = new BootstrapItem("test");
34              item.process(() -> {
35                  System.out.print("Test plugin loaded.");
36              });
37              bootstrap.add(item);
38          } catch (Exception e) {
39              throw new PluginException("Could not load plugin", e);
40          }
41      }
42  }
