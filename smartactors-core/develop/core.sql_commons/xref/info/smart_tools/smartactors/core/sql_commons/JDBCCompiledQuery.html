7   import java.sql.PreparedStatement;
8   import java.sql.SQLException;
9   import java.util.List;
10  
11  /**
12   * Implementation of {@link CompiledQuery} wrapping the {@link PreparedStatement}.
13   */
14  public class JDBCCompiledQuery implements CompiledQuery {
15      private final PreparedStatement preparedStatement;
16  
17      public JDBCCompiledQuery(final PreparedStatement preparedStatement) {
18          this.preparedStatement = preparedStatement;
19      }
20  
21      public PreparedStatement getPreparedStatement() {
22          return preparedStatement;
23      }
24  
25      public void setParameters(final List<SQLQueryParameterSetter> parameterSetters)
26              throws SQLException, QueryBuildException {
27          int index = 1;
28          for (SQLQueryParameterSetter setter : parameterSetters) {
29              index = setter.setParameters(this.preparedStatement, index);
30          }
31      }
32  }
