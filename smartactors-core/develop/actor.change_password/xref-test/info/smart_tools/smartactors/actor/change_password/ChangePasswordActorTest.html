135 
136         Object poolGuardObject = mock(Object.class);
137         when(poolGuard.getObject()).thenReturn(poolGuardObject);

141         when(message.getPassword()).thenReturn(testPassword);
142 
143         IObject searchQuery = mock(IObject.class);
144         ChangePasswordActor changePasswordActorSpy = spy(testActor);
145 
146         doReturn(searchQuery).when(changePasswordActorSpy, "prepareQueryParams", message);
147 
148         LinkedList<IObject> items = mock(LinkedList.class);

150 
151         ITask searchTask = mock(ITask.class);
152

154         when(Keys.getOrAdd("db.collection.search")).thenReturn(searchTaskKey);
155 
156         ArgumentCaptor<IAction> searchActionArgumentCaptor = ArgumentCaptor.forClass(IAction.class);
157 
158         when(IOC.resolve
159                         (eq(searchTaskKey),
160                         eq(poolGuardObject),
161                         eq(collectionName),
162                         eq(searchQuery),
163                         any(IAction.class))

165 
166         doNothing().when(changePasswordActorSpy, "validateSearchResult", items, message);
167 
168         IObject user = mock(IObject.class);
169         when(items.get(0)).thenReturn(user);
170

172         when(passwordEncoder.encode(testPassword)).thenReturn(encodedPassword);
173 
174         IKey upsertTaskKey = mock(IKey.class);
175         when(Keys.getOrAdd("db.collection.upsert")).thenReturn(upsertTaskKey);
176 
177         ITask upsertTask = mock(ITask.class);
178         when(IOC.resolve(
179                 upsertTaskKey,
180                 poolGuardObject,
181                 collectionName,
182                 user
183         )).thenReturn(upsertTask);
184 
185         changePasswordActorSpy.changePassword(message);
186 
187         verifyNew(PoolGuard.class).withArguments(connectionPool);
188 
189         verify(message).getUserId();
190         verify(message, times(2)).getPassword();
191 
192         verifyNew(LinkedList.class).withNoArguments();
193 
194         verifyStatic();
195         Keys.getOrAdd("db.collection.search");
196 
197         verifyStatic();
198         IOC.resolve
199                 (eq(searchTaskKey),
200                         eq(poolGuardObject),
201                         eq(collectionName),
202                         eq(searchQuery),
203                         searchActionArgumentCaptor.capture());
204 
205         verify(searchTask).execute();
206 
207         verify(items).get(0);
208 
209         verify(passwordField).out(user, encodedPassword);
210 
211         verifyStatic();
212         Keys.getOrAdd("db.collection.upsert");
213

215         IOC.resolve(
216                 upsertTaskKey,
217                 poolGuardObject,

223         verify(poolGuard).close();
