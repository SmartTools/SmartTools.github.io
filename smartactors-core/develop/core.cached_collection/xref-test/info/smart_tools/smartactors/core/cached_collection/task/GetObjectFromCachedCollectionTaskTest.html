78  
79          IObject query = mock(IObject.class);

81          when(keyValueField.in(query)).thenReturn("keyValue");
82  
83          IObject queryForNestedTask = mock(IObject.class);
84          IKey keyIObject = mock(IKey.class);
85          when(Keys.getOrAdd(IObject.class.toString())).thenReturn(keyIObject);
86          when(IOC.resolve(keyIObject)).thenReturn(queryForNestedTask);
87  
88          IField criteriaEqualsKeyField = mock(IField.class);
89          when(IOC.resolve(mockKeyField, "keyName/$eq/keyValue")).thenReturn(criteriaEqualsKeyField);
90  
91          testTask.prepare(query);

93          verify(targetTask).prepare(queryForNestedTask);
