17  /**
18   * Plugin for Authentication actor
19   */
20  public class AuthenticationActorPlugin implements IPlugin {
21  
22      private final IBootstrap<IBootstrapItem<String>> bootstrap;
23  
24      public AuthenticationActorPlugin(final IBootstrap<IBootstrapItem<String>> bootstrap) {
25          this.bootstrap = bootstrap;
26      }
27  
28      /**
29       * Load the plugin for Authentication actor
30       * @throws PluginException Throw when plugin can't be load
31       */
32      @Override
33      public void load() throws PluginException {
34          try {
35              IKey cachedCollectionKey = Keys.getOrAdd(AuthenticationActor.class.toString());
36              IBootstrapItem<String> item = new BootstrapItem("AuthenticationActorPlugin");
37  
38              item.after("IOC").process(() -> {
39                  try {
40                      IOC.register(cachedCollectionKey, new CreateNewInstanceStrategy(
41                              (args) -> {
42                                  try {
43                                      return new AuthenticationActor();
44                                  } catch (Exception e) {
45                                      throw new RuntimeException(e);
46                                  }
47                              }));
48                  } catch (RegistrationException | InvalidArgumentException e) {
49                      throw new ActionExecuteException(e);
50                  }
51              });
52              bootstrap.add(item);
53          } catch (ResolutionException | InvalidArgumentException e) {
54              throw new PluginException("Can't load AuthenticationActor plugin", e);
55          }
56      }
57  }
