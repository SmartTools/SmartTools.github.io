46      @Test
47      public void checkLoadExecution()
48              throws Exception {
49          Checker checker = new Checker();
50          checker.item = new BootstrapItem("test");
51          IBootstrap<IBootstrapItem<String>> bootstrap = mock(IBootstrap.class);
52          List<IBootstrapItem<String>> itemList = new ArrayList<>();
53          doAnswer(new Answer<Void>() {
54              public Void answer(InvocationOnMock invocation) {
55                  Object[] args = invocation.getArguments();
56                  checker.item = (IBootstrapItem<String>) args[0];
57                  itemList.add(checker.item);
58                  return null;
59              }
60          })
61                  .when(bootstrap)
62                  .add(any(IBootstrapItem.class));
63          IPlugin plugin = new LoadScopeProvider(bootstrap);
64          plugin.load();
65          assertEquals(itemList.size(), 1);
66          IBootstrapItem<String> item = itemList.get(0);
67          item.executeProcess();
68          reset(bootstrap);
69      }
70
