101     @Test
102     public void checkClassGenerationWithUsingJarFile()
103             throws Exception {
104 
105         String pathToJar = this.getClass().getClassLoader().getResource("ifield_name.jar").getFile();
106         JarFile jarFile = new JarFile(pathToJar);
107         Enumeration<JarEntry> e = jarFile.entries();
108 
109         URL[] urls = { new URL("jar:file:" + pathToJar+"!/") };
110         URLClassLoader cl = URLClassLoader.newInstance(urls);
111 
112         while (e.hasMoreElements()) {
113             JarEntry je = e.nextElement();
114             if(je.isDirectory() || !je.getName().endsWith(".class")){
115                 continue;
116             }
117             // -6 because of .class
118             String className = je.getName().substring(0,je.getName().length()-6);
119             className = className.replace('/', '.');
120             Class c = cl.loadClass(className);
121 
122         }
123 
124         ClassGenerator classGenerator = new ClassGenerator(cl);
125         String testSample = "package info.smart_tools.smartactors.core.test_class;\n" +
126                 "import info.smart_tools.smartactors.core.class_generator_java_compile_api.TestInterface;\n" +
127                 "import info.smart_tools.smartactors.core.ifield_name.IFieldName;\n" +
128                 "public class TestClass implements TestInterface, IFieldName {\n" +
129                 "    private int a;\n" +
130                 "    public Integer getA() {\n" +
131                 "        return a;\n" +
132                 "    }\n" +
133                 "}\n";
134         Class newClass = classGenerator.generate(testSample);
135         assertNotNull(newClass);
136     }
