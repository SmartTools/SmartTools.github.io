114     @Test
115     public void test_messageProcessingPerformance()
116             throws Exception {
117         final Thread mainThread = Thread.currentThread();
118         final AtomicLong startNanoTime = new AtomicLong();
119         final AtomicLong deltaTime = new AtomicLong();
120         final AtomicBoolean done = new AtomicBoolean(false);
121         ConcurrentMap<Long, Long> threadUseCount = new ConcurrentHashMap<>();
122 
123         IMessage messageMock = mock(IMessage.class);
124         IObject contextMock = mock(IObject.class);
125 
126         IMessageReceiver countReceiver = (mp) -> {
127             Long tid = Thread.currentThread().getId();
128             long n = threadUseCount.computeIfAbsent(tid, l -> 0L) + 1;
129             threadUseCount.put(tid, n);
130         };
131 
132         IMessageReceiver dummyReceiver = (mp) -> {};
133 
134         ITask putTask = () -> {
135             while (!mainThread.isInterrupted() && !Thread.interrupted()) {
136                 try {
137                     new MessageProcessor(taskQueue, new MessageProcessingSequence(4, standardChain), new DSObject()).process(messageMock, contextMock);
138                     Thread.sleep(PUT_INTERVAL);
139                 } catch (InterruptedException e) {
140                     Thread.currentThread().interrupt();
141                 } catch (InvalidArgumentException | ResolutionException | ChangeValueException e) {}
142             }
143         };
144 
145         for (int i = 0; i < PUT_TASKS; i++) {
146             taskQueue.put(putTask);
147         }
148 
149         IMessageReceiver countStartReceiver = (mp) -> {
150             startNanoTime.set(System.nanoTime());
151         };
152 
153         IMessageReceiver countEndReceiver = (mp) -> {
154             deltaTime.set(System.nanoTime() - startNanoTime.get());
155             System.out.println(MessageFormat.format("Messages handled in {0}ns ({1}s)", deltaTime, 0.000000001*(double)deltaTime.get()));
156             done.set(true);
157             synchronized (done) {
158                 done.notifyAll();
159             }
160         };
161 
162         standardChain = new ImmutableReceiverChain("standard",
163                 new IMessageReceiver[] {
164 //                        dummyReceiver, dummyReceiver, dummyReceiver, dummyReceiver,
165 //                        dummyReceiver, dummyReceiver, dummyReceiver, dummyReceiver,
166                         dummyReceiver, dummyReceiver, dummyReceiver, dummyReceiver,
167                         dummyReceiver, dummyReceiver, dummyReceiver, countReceiver},
168                 new IObject[8],
169                 new HashMap<>());
170 
171         IReceiverChain countStartChain = new ImmutableReceiverChain("countStart",
172                 new IMessageReceiver[] {dummyReceiver, countStartReceiver, countReceiver},
173                 new IObject[3],
174                 new HashMap<>());
175 
176         IReceiverChain countEndChain = new ImmutableReceiverChain("countEnd",
177                 new IMessageReceiver[] {
178                         dummyReceiver, dummyReceiver, dummyReceiver, dummyReceiver,
179                         dummyReceiver, dummyReceiver, dummyReceiver, dummyReceiver,
180                         dummyReceiver, dummyReceiver, dummyReceiver, dummyReceiver,
181                         dummyReceiver, dummyReceiver, dummyReceiver, dummyReceiver,
182                         dummyReceiver, dummyReceiver, dummyReceiver, dummyReceiver,
183                         countEndReceiver},
184                 new IObject[21],
185                 new HashMap<>());
186 
187         new MessageProcessor(taskQueue, new MessageProcessingSequence(4, countStartChain), new DSObject()).process(messageMock, contextMock);
188 
189         for (int i = 0; i < PAYLOAD_MESSAGES; i++) {
190             new MessageProcessor(taskQueue, new MessageProcessingSequence(4, standardChain), new DSObject()).process(messageMock, contextMock);
191         }
192 
193         new MessageProcessor(taskQueue, new MessageProcessingSequence(4, countEndChain), new DSObject()).process(messageMock, contextMock);
194 
195         dispatcher.start();
196 
197         synchronized (done) {
198             while (!done.get()) {
199                 done.wait();
200             }
201         }
202 
203         long total = 0;
204 
205         for (Long tid : threadUseCount.keySet()) {
206             long nTasks = threadUseCount.get(tid);
207             System.out.println(MessageFormat.format("T#{0}\t{1} messages", tid, nTasks));
208             total += nTasks;
209         }
210 
211         System.out.println(MessageFormat.format("Total:\t{0} messages", total));
212         System.out.println();
213         System.out.println(MessageFormat.format("Bandwidth:\t{0} messages/second",
214                 ((double)total)/(0.000000001*(double)deltaTime.get())));
215     }

217     @Test
218     public void test_messageProcessingWithChainChoicePerformance()
219             throws Exception {
220         final Thread mainThread = Thread.currentThread();
221         final AtomicLong startNanoTime = new AtomicLong();
222         final AtomicLong deltaTime = new AtomicLong();
223         final AtomicBoolean done = new AtomicBoolean(false);
224         ConcurrentMap<Long, Long> threadUseCount = new ConcurrentHashMap<>();
225 
226         IMessage messageMock = mock(IMessage.class);
227         IObject contextMock = mock(IObject.class);
228 
229         IOC.register(IOC.resolve(IOC.getKeyForKeyStorage(), "chain_id"),
230                 new ResolveByNameIocStrategy(objects -> String.valueOf(objects[0])));
231 
232         final IFieldName targetNameFieldName = IOC.resolve(IOC.resolve(IOC.getKeyForKeyStorage(), IFieldName.class.getCanonicalName()), "target");
233 
234         IOC.register(IOC.resolve(IOC.getKeyForKeyStorage(), "receiver_id_from_iobject"),
235                 new IResolveDependencyStrategy() {
236                     @Override
237                     public <T> T resolve(Object... args) throws ResolveDependencyStrategyException {
238                         try {
239                             return (T)String.valueOf(((IObject)args[0]).getValue(targetNameFieldName));
240                         } catch (ReadValueException | InvalidArgumentException e) {
241                             throw new ResolveDependencyStrategyException(e);
242                         }
243                     }
244                 });
245 
246         IOC.register(IOC.resolve(IOC.getKeyForKeyStorage(), IReceiverChain.class.toString()),
247                 new ImmutableReceiverChainResolutionStrategy());
248 
249         IOC.register(IOC.resolve(IOC.getKeyForKeyStorage(), IObject.class),
250                 new CreateNewInstanceStrategy(objects -> new DSObject()));
251 
252         IObject prepareTarget = new DSObject("{\"target\":\"prepare\"}");
253         IObject callTarget = new DSObject("{\"target\":\"call\"}");
254         IObject call2Target = new DSObject("{\"target\":\"call2\"}");
255         IObject startMeasureTarget = new DSObject("{\"target\":\"startMeasure\"}");
256         IObject endMeasureTarget = new DSObject("{\"target\":\"endMeasure\"}");
257         IObject dummyTarget = new DSObject("{\"target\":\"dummy\"}");
258         IObject countTarget = new DSObject("{\"target\":\"count\"}");
259 
260         IObject mainChainDesc = new DSObject();
261         IObject measureStartChainDesc = new DSObject();
262         IObject measureEndChainDesc = new DSObject();
263         IObject payloadChainDesc = new DSObject();
264         IObject innerPayloadChainDesc = new DSObject();
265 
266         IFieldName exceptionalFieldName = IOC.resolve(IOC.resolve(IOC.getKeyForKeyStorage(), IFieldName.class.getCanonicalName()), "exceptional");
267         IFieldName pathFieldName = IOC.resolve(IOC.resolve(IOC.getKeyForKeyStorage(), IFieldName.class.getCanonicalName()), "steps");
268 
269         // Main chain
270         mainChainDesc.setValue(exceptionalFieldName, new ArrayList<>());
271         mainChainDesc.setValue(pathFieldName, new ArrayList<>());
272         ((List)mainChainDesc.getValue(pathFieldName)).add(prepareTarget);
273         ((List)mainChainDesc.getValue(pathFieldName)).add(callTarget);
274         ((List)mainChainDesc.getValue(pathFieldName)).add(countTarget);
275 
276         // Measure start chain
277         measureStartChainDesc.setValue(exceptionalFieldName, new ArrayList<>());
278         measureStartChainDesc.setValue(pathFieldName, new ArrayList<>());
279         ((List)measureStartChainDesc.getValue(pathFieldName)).add(dummyTarget);
280         ((List)measureStartChainDesc.getValue(pathFieldName)).add(startMeasureTarget);
281 
282         // Measure end chain
283         measureEndChainDesc.setValue(exceptionalFieldName, new ArrayList<>());
284         measureEndChainDesc.setValue(pathFieldName, new ArrayList<>());
285         for (int i = 0; i < 20; i++) {
286             ((List)measureEndChainDesc.getValue(pathFieldName)).add(dummyTarget);
287         }
288         ((List)measureEndChainDesc.getValue(pathFieldName)).add(endMeasureTarget);
289 
290         // Payload chain
291         payloadChainDesc.setValue(exceptionalFieldName, new ArrayList<>());
292         payloadChainDesc.setValue(pathFieldName, new ArrayList<>());
293         for (int i = 0; i < 3; i++) {
294             ((List)payloadChainDesc.getValue(pathFieldName)).add(call2Target);
295         }
296 
297         // Inner payload chain
298         innerPayloadChainDesc.setValue(exceptionalFieldName, new ArrayList<>());
299         innerPayloadChainDesc.setValue(pathFieldName, new ArrayList<>());
300         for (int i = 0; i < 2; i++) {
301             ((List)innerPayloadChainDesc.getValue(pathFieldName)).add(dummyTarget);
302         }
303 
304         // Router & chain storage
305         IRouter router = new MapRouter(new ConcurrentHashMap<>());
306         IChainStorage storage = new ChainStorage(new ConcurrentHashMap<>(), router);
307 
308         // Receivers
309         final AtomicLong messageCounter = new AtomicLong(0);
310 
311         router.register("prepare", (mp) -> {
312             long i = messageCounter.getAndIncrement();
313             String target = null;
314 
315             if (i == 0) {
316                 target = "measureStart";
317             } else if (i == PAYLOAD_MESSAGES+1) {
318                 target = "measureEnd";
319             } else {
320                 target = "payload";
321             }
322 
323             try {
324                 mp.getMessage().setValue(targetNameFieldName, target);
325             } catch (ChangeValueException | InvalidArgumentException e) {
326                 throw new MessageReceiveException(e);
327             }
328         });
329 
330         router.register("call", new ChainCallReceiver(storage, messageProcessor -> {
331             try {
332                 return messageProcessor.getMessage().getValue(targetNameFieldName);
333             } catch (ReadValueException | InvalidArgumentException e) {
334                 throw new ChainChoiceException("Couldn't choose chain.", e);
335             }
336         }));
337 
338         router.register("call2", new ChainCallReceiver(storage, messageProcessor ->
339             "innerPayload"));
340 
341         router.register("count", (mp) -> {
342             Long tid = Thread.currentThread().getId();
343             long n = threadUseCount.computeIfAbsent(tid, l -> 0L) + 1;
344             threadUseCount.put(tid, n);
345         });
346 
347         router.register("dummy", (mp) -> { });
348 
349         router.register("startMeasure", (mp) -> {
350             startNanoTime.set(System.nanoTime());
351         });
352 
353         router.register("endMeasure", (mp) -> {
354             deltaTime.set(System.nanoTime() - startNanoTime.get());
355             System.out.println(MessageFormat.format("Messages handled in {0}ns ({1}s)", deltaTime, 0.000000001*(double)deltaTime.get()));
356             done.set(true);
357             synchronized (done) {
358                 done.notifyAll();
359             }
360         });
361 
362         // Register chins
363         storage.register("main", mainChainDesc);
364         storage.register("measureStart", measureStartChainDesc);
365         storage.register("measureEnd", measureEndChainDesc);
366         storage.register("payload", payloadChainDesc);
367         storage.register("innerPayload", innerPayloadChainDesc);
368 
369         IReceiverChain mainChain = storage.resolve("main");
370 
371         //
372 
373         for (int i = 0; i < PAYLOAD_MESSAGES + MEASURE_MESSAGES; i++) {
374             new MessageProcessor(taskQueue, new MessageProcessingSequence(5, mainChain), new DSObject())
375                     .process(new DSObject(), new DSObject());
376         }
377 
378         dispatcher.start();
379 
380         synchronized (done) {
381             while (!done.get()) {
382                 done.wait();
383             }
384         }
385 
386         long total = 0;
387 
388         for (Long tid : threadUseCount.keySet()) {
389             long nTasks = threadUseCount.get(tid);
390             System.out.println(MessageFormat.format("T#{0}\t{1} messages", tid, nTasks));
391             total += nTasks;
392         }
393 
394         System.out.println(MessageFormat.format("Total:\t{0} messages", total));
395         System.out.println();
396         System.out.println(MessageFormat.format("Bandwidth:\t{0} messages/second",
397                 ((double)total)/(0.000000001*(double)deltaTime.get())));
398     }
