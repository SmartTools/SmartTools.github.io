3   import info.smart_tools.smartactors.core.itask.ITask;
4   import info.smart_tools.smartactors.core.itask.exception.TaskExecutionException;
5   
6   /**
7    * The task that is started by {@link TaskDispatcher}. It takes tasks from queue and executes them trying to start
8    * itself in new threads (from thread pool).
9    */
10  class ExecutionTask implements ITask {
11      private final TaskDispatcher taskDispatcher;
12  
13      /**
14       * The constructor.
15       *
16       * @param taskDispatcher the task dispatcher instance that created this task.
17       */
18      ExecutionTask(final TaskDispatcher taskDispatcher) {
19          this.taskDispatcher = taskDispatcher;
20      }
21  
22      @Override
23      public void execute() throws TaskExecutionException {
24          taskDispatcher.notifyThreadStart();
25  
26          try {
27              while (!Thread.interrupted()) {
28                  ITask task = taskDispatcher.getTaskQueue().tryTake();
29  
30                  if (null == task) {
31                      return;
32                  }
33  
34                  taskDispatcher.notifyTaskTaken();
35  
36                  taskDispatcher.getThreadPool().tryExecute(this);
37  
38                  task.execute();
39              }
40          } finally {
41              taskDispatcher.notifyThreadStop();
42          }
43      }
44  }
