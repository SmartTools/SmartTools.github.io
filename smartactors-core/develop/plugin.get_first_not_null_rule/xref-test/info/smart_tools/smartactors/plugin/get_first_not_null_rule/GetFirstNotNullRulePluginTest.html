36      public void before() {

38          mockStatic(Keys.class);

45      @Test
46      public void MustCorrectLoadPlugin() throws Exception {
47          BootstrapItem bootstrapItem = mock(BootstrapItem.class);
48          whenNew(BootstrapItem.class).withArguments("GetFirstNotNullRulePlugin").thenReturn(bootstrapItem);
49  
50          when(bootstrapItem.after(anyString())).thenReturn(bootstrapItem);
51          when(bootstrapItem.before(anyString())).thenReturn(bootstrapItem);
52  
53          targetPlugin.load();
54  
55          verifyNew(BootstrapItem.class).withArguments("GetFirstNotNullRulePlugin");
56  
57          ArgumentCaptor<IPoorAction> actionArgumentCaptor = ArgumentCaptor.forClass(IPoorAction.class);
58  
59          verify(bootstrapItem).after("IOC");
60          verify(bootstrapItem).after("wds_object");
61          verify(bootstrapItem).before("starter");
62          verify(bootstrapItem).process(actionArgumentCaptor.capture());
63  
64          verify(bootstrap).add(bootstrapItem);
65  
66          IKey strategyKey = mock(IKey.class);
67          when(Keys.getOrAdd(IResolveDependencyStrategy.class.getCanonicalName())).thenReturn(strategyKey);
68  
69          GetFirstNotNullRule targetObject = mock(GetFirstNotNullRule.class);
70          whenNew(GetFirstNotNullRule.class).withNoArguments().thenReturn(targetObject);
71  
72          actionArgumentCaptor.getValue().execute();
73  
74          verifyStatic();
75          Keys.getOrAdd(IResolveDependencyStrategy.class.getCanonicalName());
76  
77          verifyNew(GetFirstNotNullRule.class).withNoArguments();
78  
79          verifyStatic();
80          IOC.resolve(strategyKey, "getFirstNotNullRule", targetObject);
81      }
