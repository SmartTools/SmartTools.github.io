18  /**
19   * Actor for creating user
20   */
21  public class CreateUserActor {
22      private IPool connectionPool;
23      private String COLLECTION_NAME = "user_account";
24  
25      /**
26       * Constructor
27       * @param params the actors params
28       * @throws InvalidArgumentException Throw when can't read some value from message or resolving key or dependency is throw exception
29       */
30      public CreateUserActor(final IObject params) throws InvalidArgumentException {
31          try {
32              ConnectionOptions connectionOptions = IOC.resolve(Keys.getOrAdd("PostgresConnectionOptions"));
33              this.connectionPool = IOC.resolve(Keys.getOrAdd("PostgresConnectionPool"), connectionOptions);
34          } catch (ResolutionException e) {
35              throw new InvalidArgumentException("Can't get key or resolve dependency", e);
36          }
37      }
38  
39      /**
40       * Create a new user in collection
41       * @param message the message
42       * @throws TaskExecutionException Throw when can't get user or upsert his
43       */
44      public void create(final MessageWrapper message) throws TaskExecutionException {
45          try {
46              IObject user = message.getUser();
47  
48              try (IPoolGuard poolGuard = new PoolGuard(connectionPool)) {
49                  ITask searchTask = IOC.resolve(
50                          Keys.getOrAdd("db.collection.upsert"),
51                          poolGuard.getObject(),
52                          COLLECTION_NAME,
53                          user
54                  );
55                  searchTask.execute();
56              }
57          } catch (PoolGuardException e) {
58              throw new TaskExecutionException("Failed to get connection", e);
59          } catch (ResolutionException e) {
60              throw new TaskExecutionException("Failed to resolve upsert task", e);
61          } catch (ReadValueException e) {
62              throw new TaskExecutionException("Failed to get user object from message", e);
63          }
64      }
65  }
