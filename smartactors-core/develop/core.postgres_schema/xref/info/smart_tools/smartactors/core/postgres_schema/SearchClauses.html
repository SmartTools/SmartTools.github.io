18  /**
19   * A set of methods to write parts of search SQL query.
20   */
21  class SearchClauses {
22  
23      /**
24       * Private constructor to avoid instantiation.
25       */
26      private SearchClauses() {
27      }
28  
29      static void writeSearchWhere(QueryStatement statement, IObject criteria) throws Exception {
30          IKey fieldNameKey = Keys.getOrAdd(IFieldName.class.getCanonicalName());
31          Writer body = statement.getBodyWriter();
32          try {
33              IFieldName filterField = IOC.resolve(fieldNameKey, "filter");
34              IObject filter = (IObject) criteria.getValue(filterField);
35              if (filter == null) {
36                  return; // no filter in the criteria, ignoring
37              }
38              body.write(" WHERE ");
39              PostgresQueryWriterResolver resolver = new PostgresQueryWriterResolver();
40              resolver.resolve(null).write(statement, resolver, null, filter);
41          } catch (ReadValueException e) {
42              // no filter in the criteria, ignoring
43          }
44      }
45  
46      static void writeSearchOrder(QueryStatement statement, IObject criteria) throws Exception {
47          IKey fieldNameKey = Keys.getOrAdd(IFieldName.class.getCanonicalName());
48          Writer body = statement.getBodyWriter();
49          try {
50              IFieldName sortField = IOC.resolve(fieldNameKey, "sort");
51              List<IObject> sortItems = (List<IObject>) criteria.getValue(sortField);
52              if (sortItems == null || sortItems.isEmpty()) {
53                  return; // no sort in the criteria, ignoring
54              }
55              body.write(" ");
56              OrderWriter order = new OrderWriter();
57              order.write(statement, sortItems);
58          } catch (ReadValueException e) {
59              // no sort in the criteria, ignoring
60          }
61      }
62  
63      static void writeSearchPaging(QueryStatement statement, IObject criteria) throws Exception {
64          IKey fieldNameKey = Keys.getOrAdd(IFieldName.class.getCanonicalName());
65          Writer body = statement.getBodyWriter();
66          try {
67              IFieldName pageField = IOC.resolve(fieldNameKey, "page");
68              IObject page = (IObject) criteria.getValue(pageField);
69              if (page == null) {
70                  return; // no page in the criteria, ignoring
71              }
72              body.write(" ");
73              Integer size;
74              Integer number;
75              try {
76                  IFieldName sizeField = IOC.resolve(fieldNameKey, "size");
77                  size = (Integer) page.getValue(sizeField);
78                  IFieldName numberField = IOC.resolve(fieldNameKey, "number");
79                  number = (Integer) page.getValue(numberField);
80              } catch (Exception e) {
81                  throw new QueryBuildException("wrong page format: " + page.serialize(), e);
82              }
83              PagingWriter paging = new PagingWriter();
84              paging.write(statement, number, size);
85          } catch (ReadValueException e) {
86              // no page in the criteria, ignoring
87          }
88      }

90  }
