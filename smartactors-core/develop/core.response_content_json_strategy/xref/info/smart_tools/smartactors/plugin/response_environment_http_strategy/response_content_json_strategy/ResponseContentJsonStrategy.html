8   import java.nio.charset.Charset;

10  /**
11   * Class for setting json content of the response
12   */
13  public class ResponseContentJsonStrategy implements IResponseContentStrategy {
14      @Override
15      public void setContent(final IObject responseObject, final IResponse response) throws SerializeException {
16          response.setContent(serializeIObjectToJson(responseObject));
17      }
18  
19      private byte[] serializeIObjectToJson(final IObject responseObject) throws SerializeException {
20          return ((String) responseObject.serialize()).getBytes(Charset.forName("UTF-8"));
21      }
22  }
