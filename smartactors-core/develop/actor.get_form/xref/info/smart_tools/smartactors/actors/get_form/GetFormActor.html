16  /**
17   * Actor that put form to message from cached collection
18   */
19  public class GetFormActor {
20      private static final String KEY_NAME = "form";
21      private ICachedCollection collection;
22  
23      /**
24       * Constructor for actor
25       * @param params the wrapper for IObject, contains collectionName
26       * @throws GetFormActorException for errors during create
27       */
28      public GetFormActor(final IObject params) throws GetFormActorException {
29          try {
30              IField field = IOC.resolve(Keys.getOrAdd(IField.class.toString()), "collectionName");
31              collection = IOC.resolve(Keys.getOrAdd(CachedCollection.class.toString()), field.in(params), KEY_NAME);
32          } catch (Exception e) {
33              throw new GetFormActorException("Can't create GetFormActor", e);
34          }
35      }
36  
37      /**
38       * Set form to message using strategy
39       * @param message the wrapper for message
40       * @throws GetFormActorException for errors during get form
41       */
42      public void getForm(final GetFormMessage message) throws  GetFormActorException {
43          try {
44              List<IObject> forms = collection.getItems(message.getFormKey());
45              IFormsStrategy strategy = IOC.resolve(Keys.getOrAdd(IFormsStrategy.class.toString()), message.getFormKey());
46              message.setForm(strategy.getForm(forms));
47  
48          } catch (Exception e) {
49              throw new GetFormActorException("Failed to get form from collection", e);
50          }
51      }
52  }
