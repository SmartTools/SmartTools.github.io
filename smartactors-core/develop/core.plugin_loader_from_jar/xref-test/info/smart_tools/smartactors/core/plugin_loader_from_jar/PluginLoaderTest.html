21      @Test
22      public void checkPluginLoaderCreation()
23              throws Exception {
24          Checker checker = new Checker();
25          ExpansibleURLClassLoader cl = new ExpansibleURLClassLoader(new URL[]{});
26          IPluginLoaderVisitor<String> visitor = mock(IPluginLoaderVisitor.class);
27          IPluginLoader pl = new PluginLoader(
28                  cl,
29                  (t) -> {
30                      try {
31                          checker.wasCalled = true;
32                      } catch (Exception e) {
33                          throw new RuntimeException("Could not create instance of IPlugin");
34                      }
35                  },
36                  visitor);
37          URL url = this.getClass().getClassLoader().getResource("test_jar_package.jar");
38          if (null == url) {
39              fail();
40          }
41          String jarPath = url.getPath();
42          pl.loadPlugin(jarPath);
43          assertTrue(checker.wasCalled);
44      }
