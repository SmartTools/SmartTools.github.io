21  public class PluginDSObject implements IPlugin {
22      private final IBootstrap<IBootstrapItem<String>> bootstrap;
23  
24      /**
25       * The constructor
26       * @param bootstrap    the bootstrap
27       */
28      public PluginDSObject(final IBootstrap<IBootstrapItem<String>> bootstrap) {
29          this.bootstrap = bootstrap;
30      }
31  
32      @Override
33      public void load() throws PluginException {
34          try {
35              IBootstrapItem<String> dsObjectItem = new BootstrapItem("iobject");
36  
37              dsObjectItem
38                      .after("IOC")
39                      .process(() -> {
40                          try {
41                              IOC.register(Keys.getOrAdd(IObject.class.getCanonicalName()),
42                                      new CreateNewInstanceStrategy(args -> {
43                                          if (args.length == 0) {
44                                              return new DSObject();
45                                          } else if (args.length == 1 && args[0] instanceof String) {
46                                              try {
47                                                  return new DSObject((String) args[0]);
48                                              } catch (InvalidArgumentException e) {
49                                                  throw new RuntimeException(e);
50                                              }
51                                          } else {
52                                              throw new RuntimeException("Invalid arguments for IObject creation.");
53                                          }
54                                      }));
55                          } catch (ResolutionException e) {
56                              throw new ActionExecuteException("Dsobject plugin can't load: can't get Dsobject key", e);
57                          } catch (InvalidArgumentException e) {
58                              throw new ActionExecuteException("Dsobject plugin can't load: can't create strategy", e);
59                          } catch (RegistrationException e) {
60                              throw new ActionExecuteException("Dsobject plugin can't load: can't register new strategy", e);
61                          }
62                      });
63  
64              bootstrap.add(dsObjectItem);
65          } catch (InvalidArgumentException e) {
66              throw new PluginException(e);
67          }
68      }
69  }
