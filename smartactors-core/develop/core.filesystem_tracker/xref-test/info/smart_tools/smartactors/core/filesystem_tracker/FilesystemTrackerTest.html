84      @Test
85      public void Should_executionOfActionPassedToTaskFactoryCauseExecutionOfAddedActions()
86              throws Exception {
87          File[] fileMocks = new File[]{mock(File.class),mock(File.class), mock(File.class)};
88          IAction<File> fileActionMock = mock(IAction.class);
89  
90          when(filenameFilterMock.accept(any(),any())).thenReturn(true);
91  
92          FilesystemTracker tracker = new FilesystemTracker(filenameFilterMock, taskFactoryMock);
93          tracker.start(directoryMock);
94  
95          ArgumentCaptor<IAction> actionCaptor = ArgumentCaptor.forClass(IAction.class);
96  
97          verify(taskFactoryMock).createRunnable(same(directoryMock), actionCaptor.capture());
98  
99          IAction<File> capturedAction = actionCaptor.getValue();
100 
101         capturedAction.execute(fileMocks[0]);
102         tracker.addFileHandler(fileActionMock);
103         verify(fileActionMock).execute(same(fileMocks[0]));
104 
105         capturedAction.execute(fileMocks[1]);
106         capturedAction.execute(fileMocks[1]);
107         verify(fileActionMock, times(1)).execute(same(fileMocks[1]));
108 
109         tracker.removeFileHandler(fileActionMock);
110         capturedAction.execute(fileMocks[2]);
111         verify(fileActionMock, never()).execute(same(fileMocks[2]));
112     }

141             throws Exception {

149

151         tracker.start(directoryMock);
152 
153         ArgumentCaptor<IAction> actionCaptor = ArgumentCaptor.forClass(IAction.class);
154 
155         verify(taskFactoryMock).createRunnable(same(directoryMock), actionCaptor.capture());
156 
157         IAction<File> capturedAction = actionCaptor.getValue();
158 
159         tracker.addErrorHandler(errorActionMock);
160         tracker.addFileHandler(fileActionMock);
161         capturedAction.execute(fileMock);
162         verify(errorActionMock).execute(same(exceptionMock));
