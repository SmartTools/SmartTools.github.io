57      @Test
58      public void Should_queryListOfExistFilesAndPollWatchServiceEvents()
59              throws Exception {
60          DirectoryStream directoryStreamMock = mock(DirectoryStream.class);
61          Iterator directoryStreamIteratorMock = mock(Iterator.class);
62          java.nio.file.Path[] filesMock = new java.nio.file.Path[] {
63                  mock(java.nio.file.Path.class),
64                  mock(java.nio.file.Path.class)
65          };
66          when(filesMock[0].toString()).thenReturn("dir/one");
67          when(filesMock[1].toString()).thenReturn("dir/two");
68  
69          when(directoryStreamIteratorMock.hasNext()).thenReturn(true, true, false);
70          when(directoryStreamIteratorMock.next()).thenReturn(filesMock[0], filesMock[1]);
71          when(directoryStreamMock.iterator()).thenReturn(directoryStreamIteratorMock);
72          when(fileSystemProviderMock.newDirectoryStream(any(), any())).thenReturn(directoryStreamMock);
73  
74          java.nio.file.Path newFileEventPathMock = mock(java.nio.file.Path.class);
75          when(newFileEventPathMock.toString()).thenReturn("new");
76          java.nio.file.Path newFileActualPathMock = mock(java.nio.file.Path.class);
77          when(newFileActualPathMock.toString()).thenReturn("dir/new");
78          when(directoryPathMock.resolve(newFileEventPathMock)).thenReturn(newFileActualPathMock);
79  
80          when(event1Mock.kind()).thenReturn((WatchEvent.Kind)StandardWatchEventKinds.ENTRY_CREATE);
81          when(event1Mock.context()).thenReturn(newFileEventPathMock);
82          when(event2Mock.kind()).thenReturn((WatchEvent.Kind)StandardWatchEventKinds.ENTRY_DELETE);
83          when(watchKeyMock.pollEvents())
84                  .thenReturn(Arrays.asList(event1Mock, event2Mock))
85                  .thenReturn(Collections.emptyList());
86  
87          when(watchServiceMock.take()).thenReturn(watchKeyMock);
88          when(watchKeyMock.reset()).thenReturn(true).thenReturn(false);
89  
90          ListenerTask task = new ListenerTask(directoryMock, actionMock, fileSystemMock);
91  
92          task.run();
93  
94          verify(actionMock, times(1)).execute(eq(new info.smart_tools.smartactors.core.path.Path("dir/one")));
95          verify(actionMock, times(1)).execute(eq(new info.smart_tools.smartactors.core.path.Path("dir/two")));
96          verify(actionMock, times(1)).execute(eq(new info.smart_tools.smartactors.core.path.Path("dir/new")));
97          verify(actionMock, times(3)).execute(any());
98      }
