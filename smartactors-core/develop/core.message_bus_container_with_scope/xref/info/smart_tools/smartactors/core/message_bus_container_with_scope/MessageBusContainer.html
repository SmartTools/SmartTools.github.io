1   package info.smart_tools.smartactors.core.message_bus_container_with_scope;
2   
3   import info.smart_tools.smartactors.core.ikey.IKey;
4   import info.smart_tools.smartactors.core.imessage_bus_container.IMessageBusContainer;
5   import info.smart_tools.smartactors.core.imessage_bus_container.exception.SendingMessageException;
6   import info.smart_tools.smartactors.core.imessage_bus_handler.IMessageBusHandler;
7   import info.smart_tools.smartactors.core.iobject.IObject;
8   import info.smart_tools.smartactors.core.scope_provider.ScopeProvider;
9   import info.smart_tools.smartactors.core.string_ioc_key.Key;
10  
11  /**
12   * Implementation of {@link IMessageBusContainer}.
13   */
14  public class MessageBusContainer implements IMessageBusContainer {
15  
16      /** Key for getting instance of {@link IMessageBusContainer} from current scope */
17      private IKey messageBusContainerKey;
18  
19      /**
20       * Default constructor
21       */
22      public MessageBusContainer() {
23          try {
24              this.messageBusContainerKey = new Key(java.util.UUID.randomUUID().toString());
25          } catch (Exception e) {
26              throw new RuntimeException("Initialization of Message bus container has been failed.");
27          }
28      }
29  
30      @Override
31      public IKey getMessageBusKey() {
32          return this.messageBusContainerKey;
33      }
34  
35      @Override
36      public void send(final IObject message) throws SendingMessageException {
37          try {
38              IMessageBusHandler handler = (IMessageBusHandler) ScopeProvider
39                      .getCurrentScope()
40                      .getValue(messageBusContainerKey);
41              handler.handle(message);
42          } catch (Throwable e) {
43              throw new SendingMessageException("Could not send message.");
44          }
45      }
46  }
