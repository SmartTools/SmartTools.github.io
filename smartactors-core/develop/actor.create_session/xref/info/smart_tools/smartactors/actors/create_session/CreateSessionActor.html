33  public class CreateSessionActor {
34      private String collectionName;
35      private IPool connectionPool;
36  
37      private IField collectionNameF;
38      private IField pageSizeF;
39      private IField pageNumberF;
40      private IField pageF;
41      private IField filterF;
42  
43      private IField sessionIdF;
44      private IField equalsF;
45      private IField authInfoF;
46  
47      /**
48       * Constructor for CreateSessionActor
49       * @param config is any configurations
50       * @throws CreateSessionException for any occurred error
51       */
52      public CreateSessionActor(final CreateSessionConfig config) throws CreateSessionException {
53          try {
54              collectionNameF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "collectionName");
55              pageSizeF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "size");
56              pageNumberF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "number");
57              pageF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "page");
58              filterF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "filter");
59  
60              sessionIdF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "sessionId");
61              equalsF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "$eq");
62              authInfoF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "authInfo");
63  
64              this.collectionName = config.getCollectionName();
65              this.connectionPool = config.getConnectionPool();
66          } catch (Exception e) {
67              throw new CreateSessionException("Can't create Actor");
68          }
69      }
70  
71      /**
72       * Check current session, if she's null create new session
73       * @param inputMessage message for checking
74       * @throws CreateSessionException Calling when throws any exception inside CreateSessionActor
75       */
76      public void resolveSession(final CreateSessionMessage inputMessage) throws CreateSessionException {
77          try {
78              String sessionId = inputMessage.getSessionId();
79              if (sessionId == null || sessionId.equals("")) {
80                  IObject authInfo = inputMessage.getAuthInfo();
81                  IObject newSession = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
82                  authInfoF.out(newSession, authInfo);
83                  inputMessage.setSession(newSession);
84              } else {
85                  try (IPoolGuard poolGuard = new PoolGuard(connectionPool)) {
86                      IObject searchQuery = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
87  
88                      //TODO:: resolve with IOC
89                      IStorageConnection connection = (IStorageConnection) poolGuard.getObject();
90                      prepareSearchQuery(searchQuery, inputMessage);
91  
92                      List<IObject> items = new LinkedList<>();
93                      ITask searchTask = IOC.resolve(
94                          Keys.getOrAdd("db.collection.search"),
95                          connection,
96                          collectionName,
97                          searchQuery,
98                          (IAction<IObject[]>) foundDocs -> {
99                              try {
100                                 items.addAll(Arrays.asList(foundDocs));
101                             } catch (Exception e) {
102                                 throw new ActionExecuteException(e);
103                             }
104                         }
105                     );
106                     searchTask.execute();
107 
108                     if (items.isEmpty()) {
109                         //TODO:: Should we create new session here?
110                         return;
111                     }
112                     IObject result = items.get(0);
113                     if (result == null) {
114                         throw new CreateSessionException("Find session is null");
115                     }
116                     inputMessage.setSession(result);
117 
118                 } catch (PoolGuardException e) {
119                     throw new CreateSessionException("Cannot get connection from pool.", e);
120                 } catch (Exception e) {
121                     throw new CreateSessionException("Error during find session by sessionId: " + inputMessage.getSessionId(), e);
122                 }
123             }
124         } catch (ReadValueException | ChangeValueException | InvalidArgumentException e) {
125             throw new CreateSessionException("Cannot create or find session by sessionId", e);
126         } catch (ResolutionException e) {
127             throw new CreateSessionException("Resolution error", e);
128         }
129     }
130 
131     private void prepareSearchQuery(final IObject searchQuery, final CreateSessionMessage inputMessage)
132             throws ChangeValueException, InvalidArgumentException, ResolutionException, ReadValueException {
133         IObject filter = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
134         IObject sessionIdObject = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
135         IObject page = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
136 
137         equalsF.out(sessionIdObject, inputMessage.getSessionId());
138         sessionIdF.out(filter, sessionIdObject);
139 
140         collectionNameF.out(searchQuery, this.collectionName);
141         pageSizeF.out(page, 1);
142         pageNumberF.out(page, 1);
143         pageF.out(searchQuery, page);
144 
145         filterF.out(searchQuery, filter);
146 
147     }
