27  /**
28   * Actor check current session, if she's null create new session
29   */
30  public class CreateSessionActor {
31      private String collectionName;
32      private IPool connectionPool;
33      //TODO:: in future will change dependency from core.db_search_task to core.db_task
34      private IObject bufferedQuery;

36      private static IField SESSION_ID_F;
37      private static IField EQUALS_F;
38  
39      static {
40          try {
41              SESSION_ID_F = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "sessionId");
42              EQUALS_F = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "$eq");
43  
44          } catch (ResolutionException e) {
45             throw new RuntimeException(e);
46          }
47      }

49      /**
50       * Constructor for CreateSessionActor
51       * @param config is any configurations
52       * @throws CreateSessionException for any occurred error

54      public CreateSessionActor(final CreateSessionConfig config) throws CreateSessionException {
55          try {
56              this.collectionName = config.getCollectionName();
57              this.connectionPool = config.getConnectionPool();

59              throw new CreateSessionException("Can't create Actor");
60          }

63      /**
64       * Check current session, if she's null create new session
65       * @param inputMessage message for checking
66       * @throws CreateSessionException Calling when throws any exception inside CreateSessionActor
67       */
68      public void resolveSession(final CreateSessionMessage inputMessage) throws CreateSessionException {
69          try {
70              String sessionId = inputMessage.getSessionId();
71              if (sessionId == null || sessionId.equals("")) {
72                  IObject authInfo = inputMessage.getAuthInfo();
73                  Session newSession = IOC.resolve(Keys.getOrAdd(Session.class.getCanonicalName()));
74                  newSession.setAuthInfo(authInfo);
75                  inputMessage.setSession(newSession);
76              } else {
77                  try (IPoolGuard poolGuard = new PoolGuard(connectionPool)) {
78                      IDatabaseTask searchTask = IOC.resolve(Keys.getOrAdd(IDatabaseTask.class.getCanonicalName()), "PSQL");
79                      IObject searchQuery = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
80  
81                      StorageConnection connection = IOC.resolve(Keys.getOrAdd(StorageConnection.class.getCanonicalName()), poolGuard.getObject());
82                      prepareSearchQuery(searchQuery, inputMessage);
83  
84                      searchTask.setConnection(connection);
85                      searchTask.prepare(searchQuery);
86                      searchTask.execute();
87  
88                      IField bufferedQueryF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "bufferedQuery");
89                      try {
90                          this.bufferedQuery = bufferedQueryF.in(searchQuery);
91                          if (bufferedQuery == null) {
92                              throw new CreateSessionException("Search Query is null.Search task didn't returned a buffered query!");
93                          }
94                      } catch (InvalidArgumentException e) {
95                          throw new CreateSessionException("Search task didn't returned a buffered query!", e);
96                      }
97  
98                      IField countSearchResultF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "countSearchResult");
99                      if (countSearchResultF.<Integer>in(searchQuery) == 0) {
100                         throw new CreateSessionException("Cannot find session by sessionId: "
101                                 + inputMessage.getSessionId()
102                         );
103                     }
104 
105                     IField searchResultField = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "searchResult");
106 
107                     IObject result = (searchResultField.<List<IObject>>in(searchQuery)).get(0);
108 
109                     IField sessionF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "session");
110                     Session fromDBSession = sessionF.in(result);
111                     if (fromDBSession == null) {
112                         throw new CreateSessionException("Find session is null");
113                     }
114                     inputMessage.setSession(fromDBSession);
115 
116                 } catch (PoolGuardException e) {
117                     throw new CreateSessionException("Cannot get connection from pool.", e);
118                 } catch (Exception e) {
119                     throw new CreateSessionException("Error during find session by sessionId: " + inputMessage.getSessionId(), e);
120                 }
121             }
122         } catch (ReadValueException | ChangeValueException e) {
123             throw new CreateSessionException("Cannot create or find session by sessionId", e);
124         } catch (ResolutionException e) {
125             throw new CreateSessionException("Error because cannot resolve Session.class", e);
126         }
127     }
128 
129     private void prepareSearchQuery(final IObject searchQuery, final CreateSessionMessage inputMessage)
130             throws ChangeValueException, InvalidArgumentException, ResolutionException, ReadValueException {
131         IObject query = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
132         IObject sessionIdObject = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
133 
134         EQUALS_F.out(sessionIdObject, inputMessage.getSessionId());
135         SESSION_ID_F.out(query, sessionIdObject);
136 
137         IField collectionNameF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "collectionName");
138         collectionNameF.out(searchQuery, this.collectionName);
139         IField pageSizeF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "pageSize");
140         pageSizeF.out(searchQuery, 1);
141         IField pageNumberF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "pageNumber");
142         pageNumberF.out(searchQuery, 1);
143         IField bufferedQueryF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "bufferedQuery");
144         bufferedQueryF.out(searchQuery, this.bufferedQuery);
145         IField criteriaF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "criteria");
146         criteriaF.out(searchQuery, query);
147 
148     }
