1   package info.smart_tools.smartactors.transformation_rules.get_header_from_request;
2   
3   import info.smart_tools.smartactors.core.iresolve_dependency_strategy.IResolveDependencyStrategy;
4   import info.smart_tools.smartactors.core.iresolve_dependency_strategy.exception.ResolveDependencyStrategyException;
5   import io.netty.handler.codec.http.FullHttpRequest;
6   import io.netty.handler.codec.http.HttpHeaders;
7   
8   /**
9    * The rule that extract header from request
10   */
11  public class GetHeaderFromRequestRule implements IResolveDependencyStrategy {
12      /**
13       * Extracts HTTP header from given HTTP request.
14       * @param args is needed parameters for resolve dependency,
15       *             an HTTP request from which data will be extracted and name of needed header
16       * @return an String with HTTP header
17       */
18      @Override
19      public <T> T resolve(final Object... args) throws ResolveDependencyStrategyException {
20          try {
21              FullHttpRequest request = (FullHttpRequest) args[0];
22              HttpHeaders headers = request.headers();
23              String headerName = (String) args[1];
24              return (T) headers.get(headerName);
25          } catch (ClassCastException e) {
26              throw new ResolveDependencyStrategyException("Some of args or header value can't be casted");
27          }
28      }
29  }
