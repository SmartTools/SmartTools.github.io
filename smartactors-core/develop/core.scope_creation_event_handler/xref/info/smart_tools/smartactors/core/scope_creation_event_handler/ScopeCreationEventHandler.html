11  /**
12   * Implementation of {@link IAction}
13   *
14   */
15  public class ScopeCreationEventHandler implements IAction<IScope> {
16  
17      /**
18       * ID of current handler
19       */
20      private IKey id;
21  
22      /**
23       * Constructor by unique handler identifier
24       * @param id unique handler identifier
25       * @throws InvalidArgumentException if any errors occurred
26       */
27      public ScopeCreationEventHandler(final IKey id)
28              throws InvalidArgumentException {
29          if (null == id) {
30              throw new InvalidArgumentException("Key should not be null.");
31          }
32          this.id = id;
33      }
34  
35      /**
36       * Add instance of {@link IStrategyContainer}
37       * @param createdScope instance of {@link IScope}
38       * @throws ActionExecuteException if any errors occurred
39       * @throws InvalidArgumentException if incoming argument are null
40       */
41      @Override
42      public void execute(final IScope createdScope)
43              throws ActionExecuteException, InvalidArgumentException {
44          if (null == createdScope) {
45              throw new InvalidArgumentException("Argument should not be null.");
46          }
47          try {
48              IStrategyContainer container = new StrategyContainer();
49              createdScope.setValue(this.id, container);
50          } catch (Exception e) {
51              throw new ActionExecuteException("Execution error.", e);
52          }
53      }
54  }
