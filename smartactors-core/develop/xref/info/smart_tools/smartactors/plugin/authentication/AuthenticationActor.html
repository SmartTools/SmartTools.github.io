8   import java.util.Objects;
9   
10  /**
11   * Actor that validate the created session
12   */
13  public class AuthenticationActor {
14  
15      /**
16       * Default constructor
17       */
18      public AuthenticationActor() {}
19  
20      /**
21       * Validate the session
22       * @param message the incoming message
23       * @throws TaskExecutionException Throw when one of parameters can't be read
24       * @throws AuthFailException Throw when auth info is null or auth info is not equal with session auth info
25       */
26      public void authenticateSession(final AuthenticationMessage message) throws TaskExecutionException, AuthFailException {
27          try {
28              String requestAuthInfo = message.getRequestUserAgent();
29              String sessionAuthInfo = message.getSessionUserAgent();
30              if (requestAuthInfo == null || !Objects.equals(requestAuthInfo, sessionAuthInfo)) {
31                  throw new AuthFailException("Failed to validate session");
32              }
33          } catch (ReadValueException e) {
34              throw new TaskExecutionException("Failed to validate session", e);
35          }
36      }
37  }
