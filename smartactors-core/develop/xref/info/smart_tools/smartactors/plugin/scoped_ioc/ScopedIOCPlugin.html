18  public class ScopedIOCPlugin implements IPlugin {
19      private IBootstrap<IBootstrapItem<String>> bootstrap;
20  
21      /**
22       * The constructor.
23       *
24       * @param bootstrap    the bootstrap
25       */
26      public ScopedIOCPlugin(final IBootstrap<IBootstrapItem<String>> bootstrap) {
27          this.bootstrap = bootstrap;
28      }
29  
30      @Override
31      public void load() throws PluginException {
32          try {
33              /* "subscribe_ioc_for_scope_creation" - create new strategy container in new scopes */
34              IBootstrapItem<String> subscribeItem = new BootstrapItem("subscribe_ioc_for_scope_creation");
35  
36              subscribeItem
37                      .before("create_system_scope")
38                      .after("scope_provider_container")
39                      .process(() -> {
40                          try {
41                              ScopeProvider.subscribeOnCreationNewScope(scope -> {
42                                  try {
43                                      scope.setValue(IOC.getIocKey(), new StrategyContainer());
44                                  } catch (Exception e) {
45                                      throw new Error(e);
46                                  }
47                              });
48                          } catch (ScopeProviderException e) {
49                              throw new ActionExecuteException("ScopedIOC plugin can't load.", e);
50                          }
51                      });
52  
53              bootstrap.add(subscribeItem);
54  
55              /* "ioc_container" - create and set container */
56              IBootstrapItem<String> containerItem = new BootstrapItem("ioc_container");
57  
58              containerItem
59                      .after("create_system_scope")
60                      .process(() -> { /* container is set by default so do nothing */ });
61  
62              bootstrap.add(containerItem);
63  
64              /* "IOC" - barrier after which the IOC should be ready to use */
65  
66              IBootstrapItem<String> iocItem = new BootstrapItem("IOC");
67  
68              iocItem
69                      .after("ioc_container")
70                      .process(() -> { });
71  
72              bootstrap.add(iocItem);
73          } catch (InvalidArgumentException e) {
74              throw new PluginException(e);
75          }
76      }
77  }
