16  /**
17   * Plugin for register GetCookieFromRequestRule in IOC
18   */
19  public class GetCookieFromRequestRulePlugin implements IPlugin {
20      private final IBootstrap<IBootstrapItem<String>> bootstrap;
21  
22      /**
23       * Constructor
24       * @param bootstrap bootstrap element
25       */
26      public GetCookieFromRequestRulePlugin(final IBootstrap<IBootstrapItem<String>> bootstrap) {
27          this.bootstrap = bootstrap;
28      }
29  
30      @Override
31      public void load() throws PluginException {
32          try {
33              IBootstrapItem<String> item = new BootstrapItem("GetCookieFromRequestRulePlugin");
34  
35              item
36                      .after("IOC")
37                      .after("wds_object")
38                      .before("starter")
39                      .process(() -> {
40                          try {
41                              IOC.resolve(
42                                      Keys.getOrAdd(IResolveDependencyStrategy.class.getCanonicalName()),
43                                      "getCookieFromRequestRule",
44                                      new GetCookieFromRequestRule()
45                              );
46                          } catch (ResolutionException e) {
47                              throw new ActionExecuteException(
48                                      "GetCookieFromRequestRule plugin can't load: can't get GetCookieFromRequestRule key", e
49                              );
50                          }
51                      });
52              bootstrap.add(item);
53          } catch (InvalidArgumentException e) {
54              throw new PluginException("Can't load GetCookieFromRequestRule plugin", e);
55          }
56      }
57  }
