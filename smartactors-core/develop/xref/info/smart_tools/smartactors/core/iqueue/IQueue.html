3   /**
4    * Blocking queue.
5    *
6    * @param <T> element type.
7    */
8   public interface IQueue <T> {
9       /**
10       * Put the item to the tail of this queue.
11       *
12       * @param item the item to put into queue
13       * @throws InterruptedException if thread is interrupted
14       */
15      void put(T item) throws InterruptedException;
16  
17      /**
18       * Take the element from the head of the queue blocking if the queue is empty.
19       *
20       * @return the element from head of queue
21       * @throws InterruptedException if thread is interrupted
22       */
23      T take() throws InterruptedException;
24  
25      /**
26       * Take the element from the head of the queue returning {@code null} if the queue is empty.
27       *
28       * @return the element from head of queue or {@code null} if queue is empty
29       */
30      T tryTake();
31  
32      /**
33       * Add a {@link Runnable} to run when new item appears in queue. The runnable will be executed immediately if there
34       * is any item in queue.
35       *
36       * @param callback the runnable to execute when new element is added to queue
37       */
38      void addNewItemCallback(Runnable callback);
39  
40      /**
41       * Remove a {@link Runnable} added by {@link #addNewItemCallback(Runnable)}.
42       *
43       * @param callback the runnable that should no more be run when new element is added to queue
44       * @see #addNewItemCallback(Runnable)
45       */
46      void removeNewItemCallback(Runnable callback);
47  }
