6   /**
7    * Implementation of {@link info.smart_tools.smartactors.core.ikey.IKey}
8    *
9    * <pre>
10   * This implementation has follow specific features:
11   * - string unique identifier;
12   * - overridden equals and hashCode method based on string property;
13   * </pre>
14   */
15  public class Key implements IKey {
16  
17      private String identifier;
18  
19      /**
20       * Constructor with string unique identifier
21       * @param identifier string unique identifier
22       * @throws InvalidArgumentException if any errors occurred
23       */
24      public Key(final String identifier)
25              throws InvalidArgumentException {
26          if (null == identifier || identifier.isEmpty()) {
27              throw new InvalidArgumentException("Value should not be empty or null.");
28          }
29          this.identifier = identifier;
30      }
31  
32      @Override
33      public boolean equals(final Object o) {
34          if (this == o) {
35              return true;
36          }
37          if (o == null || getClass() != o.getClass()) {
38              return false;
39          }
40          Key key = (Key) o;
41  
42          return !(identifier != null ? !identifier.equals(key.identifier) : key.identifier != null);
43  
44      }
45  
46      @Override
47      public int hashCode() {
48  
49          return identifier != null ? identifier.hashCode() : 0;
50      }
51  
52      @Override
53      public String toString() {
54          return identifier;
55      }
56  }
