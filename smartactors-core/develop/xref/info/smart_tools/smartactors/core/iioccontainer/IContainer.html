10  /**
11   * Interface of IOC
12   */
13  public interface IContainer {
14  
15      /**
16       * Return specific instance of {@link IKey} for container ID
17       * @return instance of {@link IKey}
18       */
19      IKey getIocKey();
20  
21      /**
22       * Return specific instance of {@link IKey} for resolve dependencies from key storage
23       * @param <T> type of returned value of inserted key
24       * @return instance of {@link IKey}
25       */
26      <T> IKey<IKey<T>> getKeyForKeyStorage();
27  
28      /**
29       * Resolve dependency by given given {@link IKey} instance and args
30       * @param key instance of {@link IKey}
31       * @param args needed parameters for resolve dependency
32       * @param <T> type of class for resolution
33       * @return instance of class with classId identifier
34       * @throws ResolutionException if resolution is impossible because of any errors
35       */
36      <T> T resolve(final IKey<T> key, final Object... args)
37              throws ResolutionException;
38  
39      /**
40       * Register new dependency by instance of {@link IKey}
41       * @param key instance of {@link IKey}
42       * @param strategy instance of {@link IResolveDependencyStrategy}
43       * @throws RegistrationException when registration is impossible because of any error
44       */
45      void register(final IKey key, final IResolveDependencyStrategy strategy)
46              throws RegistrationException;
47  
48      /**
49       * Remove dependency with given key
50       * @param key instance of {@link IKey}
51       * @throws DeletionException if any errors occurred
52       */
53      void remove(final IKey key)
54              throws DeletionException;
55  }
