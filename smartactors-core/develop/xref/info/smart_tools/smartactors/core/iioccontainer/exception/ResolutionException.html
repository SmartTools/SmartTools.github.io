3   import java.io.StringWriter;

5   /**
6    * Exception that occurs when resolution has been failed
7    */
8   public class ResolutionException extends Exception {
9   
10      /**
11       * Constructor with specific error message as argument
12       * @param message specific error message
13       */
14      public ResolutionException(final String message) {
15          super(message);
16      }
17  
18      /**
19       * Constructor with specific error message and specific cause as arguments
20       * @param message specific error message
21       * @param cause specific cause
22       */
23      public ResolutionException(final String message, final Throwable cause) {
24          super(message, cause);
25      }
26  
27      /**
28       * Constructor with specific cause as argument
29       * @param cause specific cause
30       */
31      public ResolutionException(final Throwable cause) {
32          super(cause);
33      }
34  
35      /**
36       * Generate new {@link ResolutionException} exception with advanced options
37       * @param targetClass Class type given for resolution
38       * @param args arguments given for resolution
39       * @param cause exception cause
40       * @return new ResolutionException with
41       */
42      public static ResolutionException ofResolution(final Class<?> targetClass, final Object[] args, final Throwable cause) {
43          StringWriter messageWriter = new StringWriter();
44  
45          messageWriter.write("Could not resolve dependency of class ");
46          messageWriter.write(targetClass.getName());
47          messageWriter.write(" with ");
48          messageWriter.write(String.valueOf(args.length));
49          messageWriter.write(" argument(s)");
50  
51          for (Object arg : args) {
52              messageWriter.write(" (");
53              messageWriter.write(arg.getClass().getSimpleName());
54              messageWriter.write(")'");
55              messageWriter.write(String.valueOf(arg));
56              messageWriter.write("'");
57          }
58  
59          return new ResolutionException(messageWriter.toString(), cause);
60      }
61  }
