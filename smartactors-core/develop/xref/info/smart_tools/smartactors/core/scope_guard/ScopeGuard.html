1   package info.smart_tools.smartactors.core.scope_guard;
2   
3   import info.smart_tools.smartactors.core.iscope.IScope;
4   import info.smart_tools.smartactors.core.scope_guard.exception.ScopeGuardException;
5   import info.smart_tools.smartactors.core.scope_provider.ScopeProvider;
6   
7   /**
8    * Implementation of {@link IScopeGuard}
9    * @since 1.7+
10   */
11  public class ScopeGuard implements IScopeGuard {
12  
13      /**
14       * Local storage for instance of {@link IScope}
15       */
16      private IScope previousScope;
17  
18      /**
19       * Locally saves and substitutes current instance of {@link IScope} by
20       * other
21       * @param key unique identifier for find {@link IScope}
22       * @throws  ScopeGuardException if any errors occurred
23       */
24      public ScopeGuard(final Object key)
25              throws ScopeGuardException {
26          try {
27              previousScope = ScopeProvider.getCurrentScope();
28              ScopeProvider.setCurrentScope(ScopeProvider.getScope(key));
29          } catch (Exception e) {
30              throw new ScopeGuardException("ScopeGuard could not to switch scope.", e);
31          }
32      }
33  
34      /**
35       * Sets locally saved {@link IScope} as current
36       * @throws ScopeGuardException if any errors occurred
37       */
38      public void close() throws ScopeGuardException {
39          try {
40              if (previousScope != null) {
41                  ScopeProvider.setCurrentScope(previousScope);
42              } else {
43                  throw new Exception();
44              }
45          } catch (Exception e) {
46              throw new ScopeGuardException("ScopeGuard could not restore original state.", e);
47          }
48      }
49  }
