1   package info.smart_tools.smartactors.core.message_processing;
2   
3   import info.smart_tools.smartactors.core.iioccontainer.exception.ResolutionException;
4   import info.smart_tools.smartactors.core.invalid_argument_exception.InvalidArgumentException;
5   import info.smart_tools.smartactors.core.iobject.IObject;
6   
7   /**
8    * Interface for a object responding for processing of a single message.
9    */
10  public interface IMessageProcessor {
11      /**
12       * Returns the message object.
13       *
14       * @return the message being processed
15       */
16      IObject getMessage();
17  
18      /**
19       * Returns the context object of the message being processed. Context contains all the non-serializable data related
20       * to the message.
21       *
22       * @return the context of message processing
23       */
24      IObject getContext();
25  
26      /**
27       * Returns the response object being formed for the processed message. The response object will be sent to the
28       * sender of the message being processed (and may be serialized in process of sending) so should contain only
29       * serializable data.
30       *
31       * @return the response for the message
32       */
33      IObject getResponse();
34  
35      /**
36       * Returns the {@link IMessageProcessingSequence} managing order of receivers receiving this message.
37       *
38       * @return the sequence of receivers that should receive the message
39       */
40      IMessageProcessingSequence getSequence();
41  
42      /**
43       * Start processing new message using this instance of {@link IMessageProcessor}.
44       *
45       * @param message    the message to process
46       * @param context    the context of message processing
47       * @throws InvalidArgumentException if message is {@code null}
48       * @throws InvalidArgumentException if context is {@code null}
49       * @throws ResolutionException if fails to resolve any dependency
50       */
51      void process(IObject message, IObject context)
52              throws InvalidArgumentException, ResolutionException;
53  }
