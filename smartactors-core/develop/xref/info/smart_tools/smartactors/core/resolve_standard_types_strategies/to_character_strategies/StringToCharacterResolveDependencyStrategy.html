2   
3   import info.smart_tools.smartactors.core.iresolve_dependency_strategy.IResolveDependencyStrategy;
4   import info.smart_tools.smartactors.core.iresolve_dependency_strategy.exception.ResolveDependencyStrategyException;
5   
6   /**
7    * Converts from String to Character
8    */
9   public class StringToCharacterResolveDependencyStrategy implements IResolveDependencyStrategy {
10  
11      @Override
12      public <T> T resolve(final Object... args) throws ResolveDependencyStrategyException {
13          try {
14              return (T) new Character(String.valueOf(args[0]).charAt(0));
15          } catch (Exception e) {
16              throw new ResolveDependencyStrategyException(e);
17          }
18      }
19  }
