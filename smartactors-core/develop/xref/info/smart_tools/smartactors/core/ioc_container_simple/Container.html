3   import info.smart_tools.smartactors.core.iioccontainer.IContainer;
4   import info.smart_tools.smartactors.core.iioccontainer.exception.DeletionException;
5   import info.smart_tools.smartactors.core.iioccontainer.exception.RegistrationException;
6   import info.smart_tools.smartactors.core.iioccontainer.exception.ResolutionException;
7   import info.smart_tools.smartactors.core.ikey.IKey;
8   import info.smart_tools.smartactors.core.iresolve_dependency_strategy.IResolveDependencyStrategy;
9   
10  import java.util.Map;
11  import java.util.concurrent.ConcurrentHashMap;
12  
13  /**
14   * Basic default implementation of {@link IContainer}
15   * <pre>
16   * Implementation features:
17   * - fast and simple
18   * </pre>
19   */
20  public class Container implements IContainer {
21  
22      private Map<IKey, IResolveDependencyStrategy> storage = new ConcurrentHashMap<IKey, IResolveDependencyStrategy>();
23  
24      /**
25       * Return specific instance of {@link IKey} for container ID
26       * @return instance of {@link IKey}
27       */
28      @Override
29      public IKey getIocKey() {
30          return null;
31      }
32  
33      /**
34       * Return specific instance of {@link IKey} for resolve dependencies from key storage
35       * @return instance of {@link IKey}
36       */
37      @Override
38      public IKey getKeyForKeyStorage() {
39          return null;
40      }
41  
42      /**
43       * Resolve dependency by given given {@link IKey} instance and args
44       * @param key instance of {@link IKey}
45       * @param args needed parameters for resolve dependency
46       * @param <T> type of class for resolution
47       * @return instance of class with classId identifier
48       * @throws ResolutionException if resolution is impossible because of any errors
49       */
50      @Override
51      public <T> T resolve(final IKey key, final Object ... args)
52              throws ResolutionException {
53          try {
54              IResolveDependencyStrategy strategy = storage.get(key);
55              return (T) strategy.resolve(args);
56          } catch (Exception e) {
57              throw new ResolutionException("Resolution of dependency failed.");
58          }
59      }
60  
61      /**
62       * Register new dependency by instance of {@link IKey}
63       * @param key instance of {@link IKey}
64       * @param strategy instance of {@link IResolveDependencyStrategy}
65       * @throws RegistrationException when registration is impossible because of any error
66       */
67      @Override
68      public void register(final IKey key, final IResolveDependencyStrategy strategy)
69              throws RegistrationException {
70          try {
71              storage.put(key, strategy);
72          } catch (Exception e) {
73              throw new RegistrationException("Registration of dependency failed.", e);
74          }
75      }
76  
77      /**
78       *
79       * Remove dependency with given key
80       * @param key instance of {@link IKey}
81       * @throws DeletionException if any errors occurred
82       */
83      @Override
84      public void remove(final IKey key)
85              throws DeletionException {
86          try {
87              storage.remove(key);
88          } catch (Exception e) {
89              throw new DeletionException("Deletion of dependency failed.", e);
90          }
91      }
92  }
