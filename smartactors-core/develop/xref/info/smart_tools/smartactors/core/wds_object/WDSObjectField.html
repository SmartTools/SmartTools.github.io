19  /**
20   * Implementation of {@link IField} only for {@code in} method
21   */
22  public class WDSObjectField implements IField {
23  
24      private static final String FIELD_ARGS = "args";
25      private static final String FIELD_NAME = "name";
26      private static final String LOCAL_KEYWORD = "local";
27      private static final String CONST_KEYWORD = "const";
28      private static final String DEFAULT_KEYWORD = "default";
29      private static final String SPLITTER = "\\/";
30      private static final String INNER_SETTER_STRATEGY_NAME = "wds_target_strategy";
31      private static final String INNER_GETTER_STRATEGY_NAME = "wds_getter_strategy";
32      private static final String OUTER_STRATEGY_NAME = "wds_outer_strategy";
33      private static final String SLASH = "/";
34  
35      private FieldName args;
36      private FieldName strategyName;
37      private List<IObject> rules;
38      private HashMap<String, IResolveDependencyStrategy> strategies = new HashMap<>();
39      private HashMap<String, IFunction<StrategyAndArgs, Object>> strategyExecutors = new HashMap<>();
40      private HashMap<String, IFunction<Arg, Object>> argumentsResolvers = new HashMap<>();

42      /**
43       * Constructor.
44       * Create instance of {@link WDSObjectField}
45       * @param methodRules the configuration of current method
46       * @throws InvalidArgumentException if incoming arguments are incorrect
47       */
48      public WDSObjectField(final List<IObject> methodRules)
49              throws InvalidArgumentException {
50          if (null == methodRules) {
51              throw new InvalidArgumentException("Method configuration should not be null.");
52          }
53          try {
54              initializeStrategyExecutors();
55              initializeArgumentsResolvers();
56              this.strategyName = new FieldName(FIELD_NAME);
57              this.args = new FieldName(FIELD_ARGS);
58              this.rules = methodRules;
59              for (IObject rule : rules) {
60                  String name = (String) rule.getValue(this.strategyName);
61                  strategies.put(
62                          name,
63                          null == this.strategyExecutors.get(name) ?
64                                  IOC.resolve(Keys.getOrAdd(IResolveDependencyStrategy.class.getCanonicalName()), name) :
65                                  null
66                  );
67              }
68          } catch (Throwable e) {
69              throw new InvalidArgumentException("Could not create instance of WDSObjectField." , e);
70          }
71      }

73      @Override
74      public <T> T in(final IObject env)
75              throws ReadValueException, InvalidArgumentException {
76          if (null == env) {
77              throw new InvalidArgumentException("Environment should not be null.");
78          }
79          Object value = null;
80          try {
81              for (IObject rule : this.rules) {
82                  List<String> args = (ArrayList<String>) rule.getValue(this.args);
83                  String name = (String) rule.getValue(this.strategyName);
84                  IFunction<StrategyAndArgs, Object> func = this.strategyExecutors.get(name);
85                  if (null == func) {
86                      func = this.strategyExecutors.get(OUTER_STRATEGY_NAME);
87                  }
88                  value = func.execute(
89                          new StrategyAndArgs(this.strategies.get(name), env, args, value)
90                  );
91              }
92          } catch (Exception e) {
93              throw new ReadValueException("Could not apply rules.", e);
94          }
95          if (null == value) {
96              return null;
97          }
98  
99          return (T) value;
100     }

108     @Override
109     public <T> void out(final IObject env, final T in)
110             throws ChangeValueException, InvalidArgumentException {
111         if (null == env) {
112             throw new InvalidArgumentException("Environment should not be null.");
113         }
114         Object value = in;
115         try {
116             for (IObject rule : rules) {
117                 List<String> args = (ArrayList<String>) rule.getValue(this.args);
118                 String name = (String) rule.getValue(this.strategyName);
119                 IFunction<StrategyAndArgs, Object> func = this.strategyExecutors.get(name);
120                 if (null == func) {
121                     func = this.strategyExecutors.get(OUTER_STRATEGY_NAME);
122                 }
123                 value = func.execute(
124                         new StrategyAndArgs(this.strategies.get(name), env, args, value)
125                 );
126             }
127         } catch (Exception e) {
128             throw new ChangeValueException("Could not apply rules.", e);
129         }
130     }

132     private Object getValueFromNestedIObject(final IObject source, final String location)
133             throws ReadValueException {
134         try {
135             String[] separated = location.split(SPLITTER);
136             if (separated.length == 1) {
137                 return source.getValue(IOC.resolve(Keys.getOrAdd(FieldName.class.getCanonicalName()), separated[0]));
138             }
139             return getNestedIObject(source, separated).getValue(
140                     IOC.resolve(Keys.getOrAdd(FieldName.class.getCanonicalName()), separated[separated.length - 1])
141             );
142         } catch (Throwable e) {
143             throw new ReadValueException("Could not read " + location + " from IObject", e);
144         }
145     }
146 
147     private IObject getNestedIObject(final IObject source, final String[] separatedPath)
148             throws Exception {
149         try {
150             IObject nestedObject = (IObject) source.getValue(
151                     IOC.resolve(Keys.getOrAdd(FieldName.class.getCanonicalName()), separatedPath[0])
152             );
153             for (int i = 1; i < separatedPath.length - 1; ++i) {
154                 nestedObject = (IObject) nestedObject.getValue(
155                         IOC.resolve(Keys.getOrAdd(FieldName.class.getCanonicalName()), separatedPath[i])
156                 );
157             }
158 
159             return nestedObject;
160         } catch (Throwable e) {
161             throw new ReadValueException("Could not read nested IObject.", e);
162         }
163     }

165     private void initializeStrategyExecutors() {
166         this.strategyExecutors.put(
167                 INNER_SETTER_STRATEGY_NAME,
168                 new IFunction<StrategyAndArgs, Object>() {
169                     @Override
170                     public Object execute(final StrategyAndArgs o)
171                             throws FunctionExecutionException, InvalidArgumentException {
172                         try {
173                             Object value = resolveSingleArgument(o.getArgs().get(0), o.getEnv(), o.getLocalValue());
174                             String source = o.getArgs().get(1);
175                             String[] split = source.split(SPLITTER);
176                             String fieldName = split[split.length - 1];
177                             String path = source.substring(0, source.lastIndexOf(SLASH));
178                             ((IObject) getValueFromNestedIObject(o.getEnv(), path)).setValue(
179                                     IOC.resolve(Keys.getOrAdd(FieldName.class.getCanonicalName()), fieldName), value
180                             );
181 
182                             return null;
183                         } catch (Throwable e) {
184                             throw new FunctionExecutionException("Could not execute function.", e);
185                         }
186                     }
187                 }
188         );
189         this.strategyExecutors.put(
190                 INNER_GETTER_STRATEGY_NAME,
191                 new IFunction<StrategyAndArgs, Object>() {
192                     @Override
193                     public Object execute(final StrategyAndArgs o)
194                             throws FunctionExecutionException, InvalidArgumentException {
195                         try {
196                             return resolveSingleArgument(o.getArgs().get(0), o.getEnv(), o.getLocalValue());
197                         } catch (Throwable e) {
198                             throw new FunctionExecutionException("Could not execute function.", e);
199                         }
200                     }
201                 }
202         );
203         this.strategyExecutors.put(
204                 OUTER_STRATEGY_NAME,
205                 new IFunction<StrategyAndArgs, Object>() {
206                     @Override
207                     public Object execute(final StrategyAndArgs o)
208                             throws FunctionExecutionException, InvalidArgumentException {
209                         try {
210                             return o.getStrategy().resolve(resolveArguments(
211                                     o.getArgs(), o.getEnv(), o.getLocalValue())
212                             );
213                         } catch (Throwable e) {
214                             throw new FunctionExecutionException("Could not execute function.", e);
215                         }
216                     }
217                 }
218         );
219     }

221     private void initializeArgumentsResolvers() {
222         this.argumentsResolvers.put(
223                 LOCAL_KEYWORD,
224                 new IFunction<Arg, Object>() {
225                     @Override
226                     public Object execute(final Arg o) throws FunctionExecutionException, InvalidArgumentException {
227                         return o.getLocalValue();
228                     }
229                 }
230         );
231         this.argumentsResolvers.put(
232                 CONST_KEYWORD,
233                 new IFunction<Arg, Object>() {
234                     @Override
235                     public Object execute(final Arg o) throws FunctionExecutionException, InvalidArgumentException {
236                         return o.getArg().split(SPLITTER)[1];
237                     }
238                 }
239         );
240         this.argumentsResolvers.put(
241                 DEFAULT_KEYWORD,
242                 new IFunction<Arg, Object>() {
243                     @Override
244                     public Object execute(final Arg o) throws FunctionExecutionException, InvalidArgumentException {
245                         try {
246                             return getValueFromNestedIObject(o.getEnv(), o.getArg());
247                         } catch (Throwable e) {
248                             throw new InvalidArgumentException("Could not get value from IObject", e);
249                         }
250                     }
251                 }
252         );
253     }
254 
255     private Object[] resolveArguments(
256             final List<String> arguments,
257             final IObject env,
258             final Object localValue
259     ) throws InvalidArgumentException {
260         Object[] resolvedArgs = new Object[arguments.size()];
261         for (int i = 0; i < arguments.size(); ++i) {
262             String key = arguments.get(i).split(SPLITTER)[0];
263             IFunction<Arg, Object> func = this.argumentsResolvers.get(key);
264             if (null == func) {
265                 func = this.argumentsResolvers.get(DEFAULT_KEYWORD);
266             }
267             resolvedArgs[i] = resolveSingleArgument(arguments.get(i), env, localValue);
268         }
269         return resolvedArgs;
270     }
271 
272     private Object resolveSingleArgument(final String arg, final IObject env, final Object localValue)
273             throws InvalidArgumentException {
274         String key = arg.split(SPLITTER)[0];
275         IFunction<Arg, Object> func = this.argumentsResolvers.get(key);
276         if (null == func) {
277             func = this.argumentsResolvers.get(DEFAULT_KEYWORD);
278         }
279         try {
280             return func.execute(new Arg(arg, env, localValue));
281         } catch (Throwable e) {
282             throw new InvalidArgumentException("Could not parse strategy argument.", e);
283         }
284     }
