25  /**
26   * The database task which is able to upsert documents into Postgres database.
27   */
28  public class PostgresUpsertTask implements IDatabaseTask {
29  
30      /**
31       * Connection to the database.
32       */
33      private IStorageConnection connection;
34      /**
35       * Collection where the document should be upserted.
36       */
37      private CollectionName collection;
38      /**
39       * Document to be upserted.
40       */
41      private IObject document;
42      /**
43       * Name of the ID field in the document.
44       */
45      private IFieldName idField;
46      /**
47       * Query, prepared during prepare(), to be compiled during execute().
48       */
49      private QueryStatement preparedQuery;
50  
51      private interface Executor {
52          void execute() throws TaskExecutionException;
53      }
54      /**
55       * Method of execution.
56       */
57      private Executor executeMethod;
58  
59      /**
60       * Creates the task
61       * @param connection the database connection where to perform upserts
62       */
63      public PostgresUpsertTask(final IStorageConnection connection) {
64          this.connection = connection;
65      }
66  
67      @Override
68      public void prepare(final IObject query) throws TaskPrepareException {
69          try {
70              UpsertMessage message = IOC.resolve(Keys.getOrAdd(UpsertMessage.class.getCanonicalName()), query);
71              collection = message.getCollectionName();
72              idField = IOC.resolve(
73                      Keys.getOrAdd(IFieldName.class.getCanonicalName()),
74                      String.format(PostgresSchema.ID_FIELD_PATTERN, collection.toString()));
75              document = message.getDocument();
76  
77              preparedQuery = new QueryStatement();
78  
79              try {
80                  Object id = document.getValue(idField);
81                  if (null == id) {
82                      prepareInsert();
83                  } else {
84                      prepareUpdate(id);
85                  }
86              } catch (ReadValueException e) {
87                  prepareInsert();
88              }
89          } catch (Exception e) {
90              throw new TaskPrepareException(e);
91          }
92      }
93  
94      /**
95       * Prepares the insert query.
96       */
97      private void prepareInsert() throws ResolutionException, QueryBuildException, ChangeValueException, InvalidArgumentException, SerializeException {
98          executeMethod = this::executeInsert;
99          PostgresSchema.insert(preparedQuery, collection);
100 
101         preparedQuery.pushParameterSetter((statement, index) -> {
102             String sqlDoc = null;
103             try {
104                 sqlDoc = document.serialize();
105             } catch (SerializeException e) {
106                 throw new SQLException("Cannot serialize document", e);
107             }
108             statement.setString(index++, sqlDoc);
109             return index;
110         });
111     }
112 
113     /**
114      * Retrieves the next ID for the inserting document.
115      * The ID is resolved from IOC using "db.collection.nextid" key.
116      * @return the new ID for the document
117      */
118     private Object nextId() throws ResolutionException {
119         return IOC.resolve(Keys.getOrAdd("db.collection.nextid"));
120     }
121 
122     /**
123      * Prepares the update query.
124      * @param id id of the document to be updated
125      */
126     private void prepareUpdate(final Object id) throws QueryBuildException {
127         executeMethod = this::executeUpdate;
128         PostgresSchema.update(preparedQuery, collection);
129 
130         preparedQuery.pushParameterSetter((statement, index) -> {
131             String sqlDoc = null;
132             try {
133                 sqlDoc = document.serialize();
134             } catch (SerializeException e) {
135                 throw new SQLException("Cannot serialize document", e);
136             }
137             statement.setString(index++, sqlDoc);
138             statement.setObject(index++, id);
139             return index;
140         });
141     }
142 
143     @Override
144     public void execute() throws TaskExecutionException {
145         executeMethod.execute();
146     }
147 
148     private void executeInsert() throws TaskExecutionException {
149         try {
150             Object id = nextId();
151             document.setValue(idField, id);
152 
153             JDBCCompiledQuery compiledQuery = (JDBCCompiledQuery) connection.compileQuery(preparedQuery);
154             PreparedStatement statement = compiledQuery.getPreparedStatement();
155             statement.execute();
156             connection.commit();
157         } catch (Exception e) {
158             try {
159                 document.deleteField(idField);
160                 connection.rollback();
161             } catch (Exception re) {
162                 // ignoring rollback failure
163             }
164             throw new TaskExecutionException("Insert to " + collection + " failed", e);
165         }
166     }
167 
168     private void executeUpdate() throws TaskExecutionException {
169         try {
170             JDBCCompiledQuery compiledQuery = (JDBCCompiledQuery) connection.compileQuery(preparedQuery);
171             PreparedStatement statement = compiledQuery.getPreparedStatement();
172             statement.execute();
173             connection.commit();
174         } catch (Exception e) {
175             try {
176                 connection.rollback();
177             } catch (Exception re) {
178                 // ignoring rollback failure
179             }
180             throw new TaskExecutionException("Update to " + collection + " failed", e);
181         }
182     }
