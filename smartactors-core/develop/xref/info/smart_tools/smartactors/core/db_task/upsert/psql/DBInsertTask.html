35      @Override
36      public void prepare(final IObject query) throws TaskPrepareException {
37  
38          //TODO:: replace by InsertMessage iobject wrapper
39          try {
40              UpsertMessage upsertMessage = IOC.resolve(Keys.getOrAdd(UpsertMessage.class.toString()), query);
41              String collectionName = upsertMessage.getCollectionName();
42              Writer writer = queryStatement.getBodyWriter();
43              writer.write(String.format(
44                  "INSERT INTO %s (%s) VALUES", CollectionName.fromString(collectionName).toString(), Schema.DOCUMENT_COLUMN_NAME
45              ));
46              writer.write("(?::jsonb)");
47              writer.write(String.format(" RETURNING %s AS id;", Schema.ID_COLUMN_NAME));
48  
49              this.compiledQuery = connection.compileQuery(queryStatement);
50          } catch (ResolutionException e) {
51              throw new TaskPrepareException("Error while resolving insert query statement.", e);
52          } catch (ReadValueException | ChangeValueException e) {
53              throw new TaskPrepareException("Error while get collection name.", e);
54          } catch (IOException | QueryBuildException e) {
55              throw new TaskPrepareException("Error while initialize insert query.", e);
56          } catch (StorageException e) {
57              e.printStackTrace();
58          }
59      }
60

62      public void setConnection(final StorageConnection connection) throws TaskSetConnectionException {
63          this.connection = connection;
64      }
65

67      public void execute() throws TaskExecutionException {
68  
69      }
