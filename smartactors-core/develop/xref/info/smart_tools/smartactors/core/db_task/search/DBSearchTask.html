22  /**
23   * Common a searching task executor.
24   */
25  public abstract class DBSearchTask implements IDatabaseTask {
26      /**
27       * Default constructor.
28       */
29      protected DBSearchTask() {}
30  
31      /**
32       * Executes a searching query of rows to database and pushes the result in the incoming message.
33       *
34       * @param query - a compiled executable query to database.
35       * @param message - source message with parameters for query.
36       *
37       * @throws TaskExecutionException when :
38       *                1. IOC resolution error;
39       *                2. Error change a value into {@link IObject};
40       *                3. Error during search query execution.
41       */
42      protected void execute(@Nonnull final CompiledQuery query, @Nonnull final ISearchQuery message)
43              throws TaskExecutionException {
44  
45          try {
46              ResultSet resultSet = ((JDBCCompiledQuery) query).getPreparedStatement().executeQuery();
47              List<IObject> objects = new LinkedList<>();
48  
49              while (resultSet.next()) {
50                  String jsonValue = resultSet.getString("document");
51                  IObject object;
52                  try {
53                      object = IOC.resolve(Keys.getOrAdd(IObject.class.toString()), jsonValue);
54                      IFieldName idFN = IOC.resolve(Keys.getOrAdd(IFieldName.class.toString()), "id");
55                      object.setValue(idFN, resultSet.getLong("id"));
56                  } catch (ChangeValueException e) {
57                      throw new TaskExecutionException("Could not set document's id field.", e);
58                  } catch (ResolutionException e) {
59                      throw new TaskExecutionException(e.getMessage(), e);
60                  } catch (InvalidArgumentException e) { //TODO added by AKutalev, reason: now IObject can throw InvalidArgumentException
61                      throw new TaskExecutionException("Invalid argument exception", e);
62                  }
63  
64                  objects.add(object);
65              }
66  
67              message.setSearchResult(objects);
68          } catch (SQLException e) {
69              throw new TaskExecutionException("Search query execution failed because of SQL exception.", e);
70          }
71      }
72  }
