3   import info.smart_tools.smartactors.core.iaction.IAction;
4   import info.smart_tools.smartactors.core.iaction.exception.ActionExecuteException;
5   import info.smart_tools.smartactors.core.ipath.IPath;
6   
7   import java.util.Collection;
8   
9   /**
10   * Class for storing to the queue of {@link FeatureManager}
11   */
12  class ExecutionPair {
13      private IAction<Collection<IPath>> action;
14      private Collection<IPath> path;
15  
16      /**
17       * Constructor.
18       * Creates new instance of {@link ExecutionPair} by given action and path
19       * @param action the instance of {@link IAction}
20       * @param path the collection of {@link IPath}
21       */
22      ExecutionPair(final IAction<Collection<IPath>> action, final Collection<IPath> path) {
23          this.action = action;
24          this.path = path;
25      }
26  
27      /**
28       * Execute current instance of {@link IAction} with current collection of {@link Path}
29       * @throws ActionExecuteException if execution of action has been failed
30       */
31      void execute() throws ActionExecuteException {
32          try {
33              this.action.execute(path);
34          } catch (Exception e) {
35              throw new ActionExecuteException(e);
36          }
37      }
38  }
