13  /**
14   *  Example of plugin.
15   */
16  public class MyPlugin implements IPlugin {
17  
18      private final IBootstrap<IBootstrapItem<String>> bootstrap;             // to register our BootstrapItem
19  
20      public MyPlugin(final IBootstrap<IBootstrapItem<String>> bootstrap) {   // this constructor signature is required
21          this.bootstrap = bootstrap;
22      }
23  
24      @Override
25      public void load() throws PluginException {
26          try {
27              IBootstrapItem<String> item = new BootstrapItem("MyPlugin");    // our item name
28              item.after("IOC");                                              // dependency, we need IOC
29              item.process(() -> {
30                  try {
31                      IKey key = IOC.resolve(IOC.getKeyForKeyStorage(), "new MyClass");
32                      IOC.register(
33                              key,
34                              // it's the initialization action of our plugin
35                              new CreateNewInstanceStrategy(
36                                      (args) -> new MyClass((String) args[0])
37                              )
38                      );
39                  } catch (Exception e) {
40                      throw new RuntimeException(e);
41                  }
42                  System.out.println("MyPlugin initialized");
43              });
44              bootstrap.add(item);                                            // tell server about our intents
45          } catch (Exception e) {
46              throw new PluginException("Could not load MyPlugin", e);
47          }
48          System.out.println("MyPlugin loaded");
49      }
50  
51  }
