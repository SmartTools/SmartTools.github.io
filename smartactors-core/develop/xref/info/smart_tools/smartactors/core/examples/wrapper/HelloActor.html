3   import info.smart_tools.smartactors.core.iobject.exception.ChangeValueException;
4   import info.smart_tools.smartactors.core.iobject.exception.ReadValueException;
5   
6   /**
7    * Sample actor,
8    * replies with "hello" message.
9    */
10  public class HelloActor {
11  
12      /**
13       * Sends greeting to the person with specified name.
14       * @param message a message where to get name and put greeting message
15       * @throws HelloActorException when something goes wrong
16       */
17      public void hello(final GreetingMessage message) throws HelloActorException {
18          try {
19              String name = message.getName();
20              message.setGreeting(String.format("Hello, %s!", name));
21          } catch (ReadValueException | ChangeValueException e) {
22              throw new HelloActorException(e);
23          }
24      }
25  
26  }
