5   import info.smart_tools.smartactors.core.ithread_pool.IThreadPool;

7   import java.util.concurrent.ArrayBlockingQueue;
8   import java.util.concurrent.BlockingQueue;
9   
10  /**
11   * Implementation of {@link IThreadPool}.
12   */
13  public class ThreadPool implements IThreadPool {
14      private final BlockingQueue<ThreadImpl> threadsQueue;
15  
16      /**
17       * The constructor.
18       *
19       * @param threadCount    initial count of threads.
20       */
21      public ThreadPool(final int threadCount) {
22          threadsQueue = new ArrayBlockingQueue<>(threadCount);
23  
24          for (int i = 0; i < threadCount; i++) {
25              threadsQueue.offer(new ThreadImpl(this));
26          }
27      }
28  
29      @Override
30      public boolean tryExecute(final ITask task)
31              throws TaskExecutionException {
32          ThreadImpl thread = threadsQueue.poll();
33  
34          if (null != thread) {
35              thread.execute(task);
36              return true;
37          }
38  
39          return false;
40      }
41  
42      /**
43       * Returns the thread to this pool.
44       *
45       * @param thread the thread
46       */
47      void returnThread(final ThreadImpl thread) {
48          if (!threadsQueue.offer(thread)) {
49              thread.interrupt();
50          }
51      }
52  }
