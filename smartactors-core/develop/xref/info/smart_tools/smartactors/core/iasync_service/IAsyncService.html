3   import java.util.concurrent.CompletableFuture;

5   /**
6    * A service which is capable to start and stop asynchronously.
7    *
8    * @param <TService> concrete type of service for chaining {@link CompletableFuture} calls.
9    */
10  public interface IAsyncService<TService> {
11      /**
12       * Asynchronously start the service.
13       *
14       * @return a future object which can be used for operation completion waiting.
15       * Future return value is just this instance.
16       */
17      CompletableFuture<TService> start();
18  
19      /**
20       * Asynchronously stop the service.
21       *
22       * @return a future object which can be used for operation completion waiting.
23       * Future return value is just this instance.
24       */
25      CompletableFuture<TService> stop();
26  }
