18  public class StandardConfigSectionsPlugin implements IPlugin {
19      private final IBootstrap<IBootstrapItem<String>> bootstrap;
20  
21      /**
22       * The constructor.
23       *
24       * @param bootstrap    the bootstrap
25       */
26      public StandardConfigSectionsPlugin(final IBootstrap<IBootstrapItem<String>> bootstrap) {
27          this.bootstrap = bootstrap;
28      }
29  
30      @Override
31      public void load() throws PluginException {
32          try {
33              /* "objects" section */
34              IBootstrapItem<String> objectsSectionItem = new BootstrapItem("config_section:objects");
35  
36              objectsSectionItem
37                      .after("configuration_manager")
38                      .after("router")
39                      .after("IFieldNamePlugin")
40                      .before("starter")
41                      .process(() -> {
42                          try {
43                              IConfigurationManager configurationManager =
44                                      IOC.resolve(Keys.getOrAdd(IConfigurationManager.class.getCanonicalName()));
45  
46                              configurationManager.addSectionStrategy(new ObjectsSectionProcessingStrategy());
47                          } catch (ResolutionException | InvalidArgumentException e) {
48                              throw new ActionExecuteException(e);
49                          }
50                      });
51  
52              bootstrap.add(objectsSectionItem);
53  
54              /* "maps" section */
55              IBootstrapItem<String> mapsSectionItem = new BootstrapItem("config_section:maps");
56  
57              mapsSectionItem
58                      .after("configuration_manager")
59                      .after("receiver_chains_storage")
60                      .after("receiver_chain")
61                      .after("IFieldNamePlugin")
62                      .before("starter")
63                      .process(() -> {
64                          try {
65                              IConfigurationManager configurationManager =
66                                      IOC.resolve(Keys.getOrAdd(IConfigurationManager.class.getCanonicalName()));
67  
68                              configurationManager.addSectionStrategy(new MapsSectionProcessingStrategy());
69                          } catch (ResolutionException | InvalidArgumentException e) {
70                              throw new ActionExecuteException(e);
71                          }
72                      });
73  
74              bootstrap.add(mapsSectionItem);
75  
76              /* "executor" section */
77              IBootstrapItem<String> executorSectionItem = new BootstrapItem("config_section:executor");
78  
79              executorSectionItem
80                      .after("configuration_manager")
81                      .after("IFieldNamePlugin")
82                      .before("starter")
83                      .process(() -> {
84                          try {
85                              IConfigurationManager configurationManager =
86                                      IOC.resolve(Keys.getOrAdd(IConfigurationManager.class.getCanonicalName()));
87  
88                              configurationManager.addSectionStrategy(new ExecutorSectionProcessingStrategy());
89                          } catch (ResolutionException | InvalidArgumentException e) {
90                              throw new ActionExecuteException(e);
91                          }
92                      });
93  
94              bootstrap.add(executorSectionItem);
95  
96              /* "endpoints" section */
97              IBootstrapItem<String> endpointsSectionItem = new BootstrapItem("config_section:endpoints");
98  
99              endpointsSectionItem
100                     .after("configuration_manager")
101                     .after("config_section:maps")
102                     .after("config_section:executor")
103                     .after("IFieldNamePlugin")
104                     .before("starter")
105                     .process(() -> {
106                         try {
107                             IConfigurationManager configurationManager =
108                                     IOC.resolve(Keys.getOrAdd(IConfigurationManager.class.getCanonicalName()));
109 
110                             configurationManager.addSectionStrategy(new EndpointsSectionProcessingStrategy());
111                         } catch (ResolutionException | InvalidArgumentException e) {
112                             throw new ActionExecuteException(e);
113                         }
114                     });
115 
116             bootstrap.add(endpointsSectionItem);
117 
118             /* "messageBus" section */
119             IBootstrapItem<String> messageBusItem = new BootstrapItem("config_section:messageBus");
120 
121             messageBusItem
122                     .after("configuration_manager")
123                     .after("config_section:maps")
124                     .after("config_section:executor")
125                     .after("IFieldNamePlugin")
126                     .before("starter")
127                     .process(() -> {
128                         try {
129                             IConfigurationManager configurationManager =
130                                     IOC.resolve(Keys.getOrAdd(IConfigurationManager.class.getCanonicalName()));
131 
132                             configurationManager.addSectionStrategy(new MessageBusSectionProcessingStrategy());
133                         } catch (ResolutionException | InvalidArgumentException e) {
134                             throw new ActionExecuteException(e);
135                         }
136                     });
137 
138             bootstrap.add(messageBusItem);
139         } catch (InvalidArgumentException e) {
140             throw new PluginException(e);
141         }
142     }
