13  /**
14   * Support class for convert types and resolve dependencies in
15   * @param <T> type of returning value for getters or type of value that will be set to iobject
16   */
17  public class Field<T> {
18      private IFieldName name;
19  
20      /**
21       * Constructor.
22       * Create instance of {@link Field} by given instance of {@link IFieldName}
23       * @param name instance of {@link IFieldName}
24       */
25      public Field(final IFieldName name) {
26          this.name = name;
27      }
28  
29      /**
30       * Get specific value from instance of {@link IObject} and convert it to T type
31       * @param object instance of {@link IObject}
32       * @param targetClass class to convert gotten value to
33       * @return instance of {@link T}
34       * @throws ReadValueException if any errors occurred when iobject had been reading
35       * @throws ChangeValueException if any errors occurred when iobject had been changing
36       * @throws InvalidArgumentException if incoming arguments are incorrect
37       */
38      public T from(final IObject object, final Class<T> targetClass)
39              throws ReadValueException, ChangeValueException, InvalidArgumentException {
40          if (object == null) {
41              throw new InvalidArgumentException("Input IObject in Field.from(IObject, Class) is null");
42          }
43          Object value = object.getValue(name);
44          if (value == null) {
45              return null;
46          }
47          if (value.getClass().equals(targetClass)) {
48              return (T) value;
49          }
50  
51          if (targetClass.isAssignableFrom(value.getClass())) {
52              return (T) value;
53          }
54  
55          try {
56              T convertedValue;
57              convertedValue = IOC.resolve(Keys.getOrAdd(targetClass.toString()), value);
58  
59              return convertedValue;
60          } catch (ResolutionException e) {
61              throw new InvalidArgumentException(
62                      "Passed object with value of field which is impossible to resolve to expected type",
63                      e
64              );
65          }
66      }
67  
68      /**
69       * Get specific value from instance of {@link IObject} and convert it to T type
70       * @param object instance of {@link IObject}
71       * @param strategyName if exists then indicate name of specific strategy for resolve dependency
72       * @return instance of {@link T}
73       * @throws ReadValueException if any errors occurred when iobject had been reading
74       * @throws ChangeValueException if any errors occurred when iobject had been changing
75       * @throws InvalidArgumentException if incoming arguments are incorrect
76       */
77      public T from(final IObject object, final String strategyName)
78              throws ReadValueException, ChangeValueException, InvalidArgumentException {
79          if (object == null) {
80              throw new InvalidArgumentException("Input IObject in Field.from(IObject, Class) is null");
81          }
82          Object value = object.getValue(name);
83          if (value == null) {
84              return null;
85          }
86  
87          try {
88              T convertedValue;
89              convertedValue = IOC.resolve(Keys.getOrAdd(strategyName), value);
90  
91              return convertedValue;
92          } catch (ResolutionException e) {
93              throw new InvalidArgumentException(
94                      "Passed object with value of field which is impossible to resolve to expected type",
95                      e
96              );
97          }
98      }
99  
100     /**
101      * Insert into {@link IObject} value-object that casts to target class.
102      * @param object IObject that wil be contain value
103      * @param value value object
104      * @throws ChangeValueException if any errors occurred when iobject had been changing
105      * @throws InvalidArgumentException if incoming arguments are incorrect
106      */
107     public void inject(final IObject object, final T value)
108             throws ChangeValueException, InvalidArgumentException {
109         object.setValue(name, value);
110     }
111 
112     /**
113      * Delete from input {@link IObject} field {@link Field#name}
114      * @param object input object
115      * @throws DeleteValueException if any errors occurred when iobject had been deleting
116      * @throws InvalidArgumentException if incoming arguments are incorrect
117      */
118     public void delete(final IObject object)
119             throws DeleteValueException, InvalidArgumentException {
120         object.deleteField(name);
121     }
122 }
