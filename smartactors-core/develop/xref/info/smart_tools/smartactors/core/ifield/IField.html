8   /**
9    * Support internal interface for wrappers
10   */
11  public interface IField {
12  
13      /**
14       * Apply rules and return result
15       * @param <T> the return type for {@code in} method
16       * @param env instance of {@link IObject} with data
17       * @return instance of {@link T}
18       * @throws ReadValueException if any errors occurred when iobject had been reading
19       * @throws InvalidArgumentException if incoming arguments are incorrect
20       * @throws ClassCastException when returning type doesn't match required type
21       */
22      <T> T in(final IObject env)
23              throws ReadValueException, InvalidArgumentException;
24  
25      /**
26       * Apply rules to given argument without return result
27       * the type of inserting parameters for {@code out} method
28       * @param env instance of {@link IObject} with data
29       * @param in given argument
30       * @throws ChangeValueException if any errors occurred when iObject had been changing
31       * @throws InvalidArgumentException if incoming arguments are incorrect
32       */
33      <T> void out(final IObject env, final T in)
34              throws ChangeValueException, InvalidArgumentException;
35  }
