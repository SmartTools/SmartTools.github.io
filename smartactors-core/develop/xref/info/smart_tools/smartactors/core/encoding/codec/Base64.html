5   import info.smart_tools.smartactors.core.security.encoding.encoders.EncodingException;

7   import javax.annotation.Nonnull;
8   
9   /**
10   * Codec implementation for base64 algorithm
11   */
12  public class Base64 implements ICodec {
13      private final java.util.Base64.Encoder base64Encoder;
14      private final java.util.Base64.Decoder base64Decoder;
15  
16      private Base64() {
17          this.base64Encoder = java.util.Base64.getEncoder();
18          this.base64Decoder = java.util.Base64.getDecoder();
19      }
20  
21      /**
22       * Factory method
23       * @return instance of Base64
24       */
25      public static Base64 create() {
26          return new Base64();
27      }
28  
29      @Override
30      public byte[] encode(@Nonnull final byte[] message) throws EncodingException {
31          try {
32              return base64Encoder.encode(message);
33          } catch (Exception e) {
34              throw new EncodingException("Base64 encoding has been failed because: " + e.getMessage(), e);
35          }
36      }
37  
38      @Override
39      public byte[] decode(@Nonnull final byte[] message) throws DecodingException {
40          try {
41              return base64Decoder.decode(message);
42          } catch (Exception e) {
43              throw new DecodingException("Base64 decoding has been failed because: " + e.getMessage(), e);
44          }
45      }
46  }
