18  /**
19   * The database task which is to count documents in {@link InMemoryDatabase} by multiple search criteria.
20   * The search criteria are passed in the complex IObject to the {@link #prepare(IObject)} method.
21   * <p>
22   * The example of prepare object:
23   * <pre>
24   *  {
25   *      "collectionName": CollectionName object,
26   *      "criteria": {
27   *          "filter": {
28   *              "$or": [
29   *                  { "a": { "$eq": "b" } },
30   *                  { "b": { "$gt": 42 } }
31   *              ]
32   *          }
33   *      },
34   *      "callback": IAction callback to receive count
35   *  }
36   *     </pre>
37   * </p>
38   * <p>
39   * The criteria is the IObject which contains a set of conditions and operators.
40   * Conditions joins operators together.
41   * Operators match the specified document field against the specified criteria.
42   * </p>
43   * <p>
44   * Available conditions:
45   * <ul>
46   * <li><code>$and</code> — ANDs operators and nested conditions</li>
47   * <li><code>$or</code> — ORs operators and nested conditions</li>
48   * <li><code>$not</code> — negate all nested operators and conditions, is equivalent to NOT(conditionA) AND NOT(conditionB)</li>
49   * </ul>
50   * Available operators:
51   * <ul>
52   * <li><code>$eq</code> — test for equality of the document field and the specified value</li>
53   * <li><code>$neq</code> — test for not equality</li>
54   * <li><code>$lt</code> — "less than", the document field is less than the specified value</li>
55   * <li><code>$gt</code> — "greater than", the document field is larger than the specified value</li>
56   * <li><code>$lte</code> — less or equal</li>
57   * <li><code>$gte</code> — greater or equal</li>
58   * <li><code>$isNull</code> — checks for null if the specified value is "true" or checks for not null if "false"</li>
59   * <li><code>$date-from</code> — greater or equal for datetime fields</li>
60   * <li><code>$date-to</code> — less or equal for datetime fields</li>
61   * <li><code>$in</code> — checks for equality to any of the specified values in the array</li>
62   * <li><code>$hasTag</code> — check the document field is JSON document contains the specified value as field name or value</li>
63   * </ul>
64   * </p>
65   */
66  
67  public class InMemoryDBCountTask implements IDatabaseTask {
68  
69      private IFieldName collectionFieldName;
70      private IFieldName criteriaFieldName;
71      private IFieldName callbackFieldName;
72  
73      /**
74       * Collection where the document should be counted.
75       */
76      private String collection;
77      /**
78       * Criteria to count the documents.
79       */
80      private IObject criteria;
81      /**
82       * Callback function to call when the documents are counted.
83       */
84      private IAction<Long> callback;
85  
86      /**
87       * Creates the task.
88       * @throws TaskPrepareException if cannot resolve IFieldName
89       */
90      public InMemoryDBCountTask() throws TaskPrepareException {
91          try {
92              collectionFieldName = IOC.resolve(Keys.getOrAdd(IFieldName.class.getCanonicalName()), "collectionName");
93              criteriaFieldName = IOC.resolve(Keys.getOrAdd(IFieldName.class.getCanonicalName()), "criteria");
94              callbackFieldName = IOC.resolve(Keys.getOrAdd(IFieldName.class.getCanonicalName()), "callback");
95          } catch (ResolutionException e) {
96              throw new TaskPrepareException("Failed to resolve \"IFieldName\"", e);
97          }
98      }
99  
100     @Override
101     public void prepare(final IObject query) throws TaskPrepareException {
102         try {
103             collection = (String) query.getValue(collectionFieldName);
104             criteria = (IObject) query.getValue(criteriaFieldName);
105             callback = (IAction<Long>) query.getValue(callbackFieldName);
106         } catch (ReadValueException | InvalidArgumentException e) {
107             throw new TaskPrepareException("Failed to getting values from query", e);
108         }
109     }
110 
111     @Override
112     public void execute() throws TaskExecutionException {
113         try {
114             IDatabase dataBase = IOC.resolve(Keys.getOrAdd(InMemoryDatabase.class.getCanonicalName()));
115             Long result = dataBase.count(criteria, collection);
116             callback.execute(result);
117         } catch (ResolutionException e) {
118             throw new TaskExecutionException("Failed to resolve InMemoryDatabase", e);
119         } catch (Exception e) {
120             try {
121                 throw new TaskExecutionException("Count failed: criteria = " + criteria.serialize(), e);
122             } catch (SerializeException | NullPointerException e1) {
123                 throw new TaskExecutionException("Count failed", e);
124             }
125         }
126     }
127 
128 }
