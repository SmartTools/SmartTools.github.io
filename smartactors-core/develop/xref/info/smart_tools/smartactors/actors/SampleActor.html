13  public class SampleActor {
14  
15      private Integer state = 0;
16  
17      public void transformAndPutForResponse(SampleWrapper wrapper)
18              throws SampleException {
19          try {
20              ++this.state;
21              String s = wrapper.getSomeField();
22              wrapper.setSomeValueForRequest(s + "_transformed");
23              wrapper.setCurrentActorState(this.state);
24              if (wrapper.resetState()) {
25                  this.state = 0;
26              }
27              IObject newMessage = IOC.resolve(
28                      IOC.resolve(IOC.getKeyForKeyStorage(), IObject.class.getCanonicalName())
29              );
30              IFieldName name = IOC.resolve(
31                      IOC.resolve(IOC.getKeyForKeyStorage(), IFieldName.class.getCanonicalName()),
32                      "value"
33              );
34              IFieldName nameMessageMapId = IOC.resolve(
35                      IOC.resolve(IOC.getKeyForKeyStorage(), IFieldName.class.getCanonicalName()),
36                      "messageMapId"
37              );
38              newMessage.setValue(name, s + "_sendingMessage");
39              newMessage.setValue(nameMessageMapId, "myChainOther");
40              MessageBus.send(newMessage);
41          } catch (Exception e) {
42              throw new SampleException();
43          }
44      }
45  }
