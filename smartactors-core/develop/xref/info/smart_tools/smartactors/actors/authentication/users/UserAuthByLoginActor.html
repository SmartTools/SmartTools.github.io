58      /**
59       * Default constructor for actor.
60       * Necessary for resolving from IOC.
61       *
62       * @param params - parameters for initialize actor.

68      public UserAuthByLoginActor(@Nonnull final IUserAuthByLoginParams params) throws InvalidArgumentException {
69          try {
70              checkParams(params);
71              this.collection = params.getCollection();
72              this.connectionPool = params.getConnectionPool();
73  
74              loginF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "email");
75              passwordF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "пароль");
76              equalsF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "$eq");
77  
78              this.passwordEncoder = IOC.resolve(
79                      Keys.getOrAdd("PasswordEncoder"),
80                      params.getAlgorithm(),
81                      params.getEncoder(),
82                      params.getCharset()
83              );
84          } catch (ResolutionException | ReadValueException e) {
85              throw new InvalidArgumentException(e.getMessage(), e);
86          }

89      /**
90       * Try to authenticate a some user by given login and password.
91       * If authentication is successfully sets the status "SUCCESS" into the given auth message,
92       *              else sets status "FAIL" and error message into the given auth message.
93       *
94       * @param message - message for authentication with user's login and password.
95       *                The user's login and password must not be a NULL or empty.
96       * @see IUserAuthByLoginMessage
97       *
98       * @throws AuthenticateUserException when:
99       *                1. User with given login hasn't registered in the system.
100      *                2. Given password is invalid.
101      *                3. In the system has a few users with given login.
102      *                4. In the system user with given login saved without password.
103      *                5. A wrong work of search database task : task not prepare a query.
104      */
105     public void authenticateUser(@Nonnull final IUserAuthByLoginMessage message)
106             throws AuthenticateUserException {
107         try (IPoolGuard poolGuard = new PoolGuard(connectionPool)) {
108             checkMsg(message);
109             IObject user = resolveLogin(message, poolGuard);
110             validatePassword(message, user);
111             setSuccessResponse(message);
112         } catch (PoolGuardException | ReadValueException | ChangeValueException | InvalidArgumentException e) {
113             try {
114                 setFailResponse(message, INTERNAL_ERROR_MSG);
115             } catch (ChangeValueException ignored) { }
116             throw new AuthenticateUserException(AUTH_ERROR_MSG + e.getMessage(), e);
117         }
118     }

120     private IObject resolveLogin(final IUserAuthByLoginMessage message, final IPoolGuard connectionPoolGuard)
121         throws AuthenticateUserException {
122 
123         try {
124             IObject searchQuery = prepareQueryParams(message);
125             List<IObject> items = new LinkedList<>();
126             ITask searchTask = IOC.resolve(
127                 Keys.getOrAdd("db.collection.search"),
128                 connectionPoolGuard.getObject(),
129                 collection,
130                 searchQuery,
131                 (IAction<IObject[]>) foundDocs -> {
132                     try {
133                         items.addAll(Arrays.asList(foundDocs));
134                     } catch (Exception e) {
135                         throw new ActionExecuteException(e);
136                     }
137                 }
138             );
139 
140             searchTask.execute();
141 
142             validateSearchResult(items, message);
143 
144             return items.get(0);
145         } catch (ResolutionException | TaskExecutionException | InvalidArgumentException | ChangeValueException | ReadValueException e) {
146             throw new AuthenticateUserException(AUTH_ERROR_MSG + e.getMessage(), e);
147         }
148     }
149 
150     private void validateSearchResult(final List<IObject> searchResult, final IUserAuthByLoginMessage message)
151         throws AuthenticateUserException {
152 
153         try {
154             if (searchResult.isEmpty()) {
155                 setFailResponse(message, AUTH_ERROR_RESPONSE_MSG);
156                 throw new AuthenticateUserException(AUTH_ERROR_MSG +
157                         "user with login: [" + message.getLogin() + "] doesn't exist!");
158             }
159             if (searchResult.size() > 1) {
160                 setFailResponse(message, AUTH_ERROR_RESPONSE_MSG);
161                 throw new AuthenticateUserException(AUTH_ERROR_MSG +
162                         "too many users with login: [" + message.getLogin() + "]!");
163             }
164         } catch (ReadValueException | ChangeValueException e) {
165             throw new AuthenticateUserException(AUTH_ERROR_MSG + e.getMessage(), e);
166         }
167     }
168 
169     private void validatePassword(final IUserAuthByLoginMessage message, final IObject user)
170             throws AuthenticateUserException {
171         try {
172             String password = passwordF.in(user);
173             if (password == null || password.isEmpty()) {
174                 setFailResponse(message, AUTH_ERROR_RESPONSE_MSG);
175                 throw new AuthenticateUserException(AUTH_ERROR_MSG +
176                         "user with login: [" + message.getLogin() + "] hasn't password!");
177             }
178             if (!password.equals(passwordEncoder.encode(message.getPassword()))) {
179                 setFailResponse(message, AUTH_ERROR_RESPONSE_MSG);
180                 throw new AuthenticateUserException(AUTH_ERROR_MSG +
181                         "Invalid password: [" + message.getPassword() + "] for login: [" + message.getLogin() + "]!");
182             }
183         } catch (EncodingException | InvalidArgumentException | ReadValueException | ChangeValueException e) {
184             throw new AuthenticateUserException(AUTH_ERROR_MSG + e.getMessage(), e);
185         }
186     }

188     private IObject prepareQueryParams(final IUserAuthByLoginMessage message)
189         throws ResolutionException, ChangeValueException, InvalidArgumentException, ReadValueException {
190 
191         IObject filter = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
192         IObject page = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
193         IObject searchQuery = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
194 
195         IField collectionNameF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "collectionName");
196         collectionNameF.out(searchQuery, this.collection);
197         IField pageSizeF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "size");
198         pageSizeF.out(page, 1);
199         IField pageNumberF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "number");
200         pageNumberF.out(page, 1);
201         IField pageF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "page");
202         pageF.out(searchQuery, page);
203         IField filterF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "filter");
204         filterF.out(searchQuery, filter);
205 
206         IObject loginObject = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
207         equalsF.out(loginObject, message.getLogin());
208         loginF.out(filter, loginObject);
209 
210         return searchQuery;
211     }

223     private void checkParams(final IUserAuthByLoginParams params) throws InvalidArgumentException, ReadValueException {
224         if (isNullOrEmpty(params.getCollection())) {
225             throw new InvalidArgumentException("Invalid collection name!");
226         }
227         if (isNullOrEmpty(params.getAlgorithm())) {
228             throw new InvalidArgumentException("Invalid algorithm for password encoding!");
229         }
230     }
231 
232     private void checkMsg(final IUserAuthByLoginMessage msg) throws InvalidArgumentException, ReadValueException, ChangeValueException {
233         if (isNullOrEmpty(msg.getLogin()) || isNullOrEmpty(msg.getPassword())) {
234             setFailResponse(msg, AUTH_ERROR_RESPONSE_MSG);
235             throw new InvalidArgumentException("Invalid message format!");
236         }
237     }

240         return str == null || str.isEmpty();
