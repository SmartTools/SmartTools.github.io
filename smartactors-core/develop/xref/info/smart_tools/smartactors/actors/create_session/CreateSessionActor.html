50      /**
51       * Constructor for CreateSessionActor
52       * @param config is any configurations
53       * @throws CreateSessionException for any occurred error
54       */
55      public CreateSessionActor(final IObject config) throws CreateSessionException {
56          try {
57              collectionNameF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "collectionName");
58              pageSizeF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "size");
59              pageNumberF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "number");
60              pageF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "page");
61              filterF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "filter");
62  
63              sessionIdF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "sessionId");
64              equalsF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "$eq");
65              authInfoF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "authInfo");
66  
67              cookieNameF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "name");
68              cookieValueF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "value");
69              maxAgeF = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "maxAge");
70  
71              ConnectionOptions connectionOptions = IOC.resolve(Keys.getOrAdd("PostgresConnectionOptions"));
72              this.connectionPool = IOC.resolve(Keys.getOrAdd("PostgresConnectionPool"), connectionOptions);
73              this.cookiesTTL = maxAgeF.in(config);
74              this.collectionName = collectionNameF.in(config);
75          } catch (Exception e) {
76              throw new CreateSessionException("Failed to create Actor");
77          }
78      }

80      /**
81       * Check current session, if she's null create new session
82       * @param inputMessage message for checking
83       * @throws CreateSessionException Calling when throws any exception inside CreateSessionActor
84       */
85      public void resolveSession(final CreateSessionMessage inputMessage) throws CreateSessionException {
86          try {
87              IObject session;
88              String sessionId = inputMessage.getSessionId();
89              if (sessionId == null || sessionId.equals("")) {
90                  sessionId = String.valueOf(UUID.randomUUID());
91                  IObject authInfo = inputMessage.getAuthInfo();
92                  session = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
93                  authInfoF.out(session, authInfo);
94                  sessionIdF.out(session, sessionId);
95              } else {
96                  try (IPoolGuard poolGuard = new PoolGuard(connectionPool)) {
97                      IObject searchQuery = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
98  
99                      //TODO:: resolve with IOC
100                     IStorageConnection connection = (IStorageConnection) poolGuard.getObject();
101                     prepareSearchQuery(searchQuery, inputMessage);
102 
103                     List<IObject> items = new LinkedList<>();
104                     ITask searchTask = IOC.resolve(
105                         Keys.getOrAdd("db.collection.search"),
106                         connection,
107                         collectionName,
108                         searchQuery,
109                         (IAction<IObject[]>) foundDocs -> {
110                             try {
111                                 items.addAll(Arrays.asList(foundDocs));
112                             } catch (Exception e) {
113                                 throw new ActionExecuteException(e);
114                             }
115                         }
116                     );
117                     searchTask.execute();
118 
119                     if (items.isEmpty()) {
120                         //TODO:: Should we create new session here?
121                         return;
122                     }
123                     session = items.get(0);
124                     if (session == null) {
125                         throw new CreateSessionException("Find session is null");
126                     }
127                 } catch (PoolGuardException e) {
128                     throw new CreateSessionException("Cannot get connection from pool.", e);
129                 } catch (Exception e) {
130                     throw new CreateSessionException("Error during find session by sessionId: " + inputMessage.getSessionId(), e);
131                 }
132             }
133             inputMessage.setSession(session);
134 
135             List<IObject> cookies = inputMessage.getCookies();
136             if (cookies == null) {
137                 cookies = new ArrayList<>();
138             }
139             IObject sessionCookie = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
140             cookieNameF.out(sessionCookie, "sessionId");
141             cookieValueF.out(sessionCookie, sessionId);
142             maxAgeF.out(sessionCookie, cookiesTTL);
143 
144             cookies.add(sessionCookie);
145             inputMessage.setCookies(cookies);
146         } catch (ReadValueException | ChangeValueException | InvalidArgumentException e) {
147             throw new CreateSessionException("Cannot create or find session by sessionId", e);
148         } catch (ResolutionException e) {
149             throw new CreateSessionException("Resolution error", e);
150         }
151     }
152 
153     private void prepareSearchQuery(final IObject searchQuery, final CreateSessionMessage inputMessage)
154             throws ChangeValueException, InvalidArgumentException, ResolutionException, ReadValueException {
155         IObject filter = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
156         IObject sessionIdObject = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
157         IObject page = IOC.resolve(Keys.getOrAdd(IObject.class.getCanonicalName()));
158 
159         equalsF.out(sessionIdObject, inputMessage.getSessionId());
160         sessionIdF.out(filter, sessionIdObject);
161 
162         collectionNameF.out(searchQuery, this.collectionName);
163         pageSizeF.out(page, 1);
164         pageNumberF.out(page, 1);
165         pageF.out(searchQuery, page);
166 
167         filterF.out(searchQuery, filter);
168 
169     }
