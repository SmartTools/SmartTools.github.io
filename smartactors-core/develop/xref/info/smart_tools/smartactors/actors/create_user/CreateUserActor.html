14  /**
15   * Actor for creating user
16   */
17  public class CreateUserActor {
18      private CachedCollection collection;
19  
20      /**
21       * Constructor
22       * @param params the actors params
23       * @throws InvalidArgumentException
24       */
25      public CreateUserActor(ActorParams params) throws InvalidArgumentException {
26          try {
27              collection = IOC.resolve(Keys.getOrAdd(CachedCollection.class.toString()), params.getCollectionName());
28          } catch (Exception e) {
29              throw new InvalidArgumentException("Failed to initialize collection", e);
30          }
31      }
32  
33      /**
34       * Create a new user in collection
35       * @param message the message
36       * @throws TaskExecutionException
37       */
38      public void create(final MessageWrapper message) throws TaskExecutionException {
39          try {
40              IObject user = message.getUser();
41              collection.upsert(user);
42          } catch (ReadValueException | UpsertCacheItemException e) {
43              throw new TaskExecutionException("Failed to create new user", e);
44          }
45      }
46  }
