2   
3   import info.smart_tools.smartactors.core.idatabase.exception.IDatabaseException;
4   import info.smart_tools.smartactors.core.iobject.IObject;
5   
6   import java.util.List;
7   
8   /**
9    * Interface for a general database
10   */
11  public interface IDatabase {
12  
13      /**
14       * Creates the collection
15       * @param collectionName name of the collection to create
16       */
17      void createCollection(String collectionName);
18  
19      /**
20       * Inserts the document to the collection
21       * @param document document to insert
22       * @param collectionName name of the collection
23       * @throws IDatabaseException if insert failed
24       */
25      void insert(final IObject document, final String collectionName) throws IDatabaseException;
26  
27      /**
28       * Updates the document in the collection
29       * @param document document to update
30       * @param collectionName name of the collection
31       * @throws IDatabaseException if update failed
32       */
33      void update(final IObject document, final String collectionName) throws IDatabaseException;
34  
35      /**
36       * Upserts (inserts or updates) the document in the collection
37       * @param document document to upsert
38       * @param collectionName name of the collection
39       * @throws IDatabaseException if upsert failed
40       */
41      void upsert(final IObject document, final String collectionName) throws IDatabaseException;
42  
43      /**
44       * Selects the document from the collection by it's ID.
45       * @param id ID of the document
46       * @param collectionName name of the collection
47       * @return the found document
48       * @throws IDatabaseException if lookup failed
49       */
50      IObject getById(final Object id, final String collectionName) throws IDatabaseException;
51  
52      /**
53       * Deletes the document from the collection.
54       * @param document document to delete
55       * @param collectionName name of the collection
56       * @throws IDatabaseException if deletion failed
57       */
58      void delete(IObject document, final String collectionName) throws IDatabaseException;
59  
60      /**
61       * Selects the documents from the collection using the specified conditions
62       * @param condition criteria of search
63       * @param collectionName name of the collection
64       * @return list of found documents
65       * @throws IDatabaseException if selection failed
66       */
67      List<IObject> select(final IObject condition, final String collectionName) throws IDatabaseException;
68  
69  }
