10   *  Example of plugin.
11   */
12  public class Plugin1 implements IPlugin {
13  
14      private final IBootstrap<IBootstrapItem<String>> bootstrap;
15  
16      public Plugin1(final IBootstrap<IBootstrapItem<String>> bootstrap) {   // this constructor signature is required
17          this.bootstrap = bootstrap;
18      }
19  
20      @Override
21      public void load() throws PluginException {
22          try {
23              IBootstrapItem<String> item = new BootstrapItem("Plugin1");
24              item.process(() -> System.out.println("Plugin1 initialized"));
25              bootstrap.add(item);
26          } catch (Exception e) {
27              throw new PluginException("Could not load Plugin1", e);
28          }
29          System.out.println("Plugin1 loaded");
30      }
31  
32  }
