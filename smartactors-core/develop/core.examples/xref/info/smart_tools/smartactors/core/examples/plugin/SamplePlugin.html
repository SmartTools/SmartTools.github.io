14  /**
15   * Example of plugin.
16   */
17  public class SamplePlugin implements IPlugin {
18  
19      private final IBootstrap<IBootstrapItem<String>> bootstrap;                 // to register our BootstrapItem
20  
21      /**
22       * Creates the plugin.
23       * This constructor is called by {@link info.smart_tools.smartactors.core.plugin_loader_from_jar.PluginLoader},
24       * so it must have one argument of type IBootstrap.
25       * @param bootstrap bootstrap where this plugin puts {@link IBootstrapItem}.
26       */
27      public SamplePlugin(final IBootstrap<IBootstrapItem<String>> bootstrap) {   // this constructor signature is required
28          this.bootstrap = bootstrap;
29      }
30  
31      @Override
32      public void load() throws PluginException {
33          try {
34              IBootstrapItem<String> item = new BootstrapItem("SamplePlugin");    // our item name
35              item.after("IOC");                                                  // dependency, we need IOC
36              item.process(() -> {
37                  try {
38                      IKey key = Keys.getOrAdd("new SampleClass");
39                      IOC.register(
40                              key,
41                              // it's the initialization action of our plugin
42                              new CreateNewInstanceStrategy(
43                                      (args) -> new SampleClass((String) args[0])
44                              )
45                      );
46                  } catch (Exception e) {
47                      throw new RuntimeException(e);
48                  }
49                  System.out.println("SamplePlugin initialized");
50              });
51              bootstrap.add(item);                                            // tell server about our intents
52          } catch (Exception e) {
53              throw new PluginException("Could not load SamplePlugin", e);
54          }
55          System.out.println("SamplePlugin loaded");
56      }
57  
58  }
