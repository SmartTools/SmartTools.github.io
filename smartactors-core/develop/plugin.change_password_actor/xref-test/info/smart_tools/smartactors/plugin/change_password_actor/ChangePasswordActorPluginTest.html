36      private ChangePasswordActorPlugin plugin;
37      private IBootstrap bootstrap;

39      @Before
40      public void setUp() throws Exception {
41  
42          mockStatic(IOC.class);
43          mockStatic(Keys.class);
44  
45          bootstrap = mock(IBootstrap.class);
46          plugin = new ChangePasswordActorPlugin(bootstrap);
47      }
48  
49  
50      @Test
51      public void ShouldCorrectLoadPlugin() throws Exception {
52  
53          BootstrapItem bootstrapItem = mock(BootstrapItem.class);
54          whenNew(BootstrapItem.class).withArguments("ChangePasswordActorPlugin").thenReturn(bootstrapItem);
55  
56          when(bootstrapItem.after(anyString())).thenReturn(bootstrapItem);
57          when(bootstrapItem.before(anyString())).thenReturn(bootstrapItem);
58  
59          plugin.load();
60  
61          verifyNew(BootstrapItem.class).withArguments("ChangePasswordActorPlugin");
62  
63          ArgumentCaptor<IPoorAction> actionArgumentCaptor = ArgumentCaptor.forClass(IPoorAction.class);
64  
65          verify(bootstrapItem).after("IOC");
66          verify(bootstrapItem).before("starter");
67          verify(bootstrapItem).process(actionArgumentCaptor.capture());
68          verify(bootstrap).add(bootstrapItem);
69  
70          IKey changePasswordActorKey = mock(IKey.class);
71          when(Keys.getOrAdd(ChangePasswordActor.class.getCanonicalName())).thenReturn(changePasswordActorKey);
72  
73          actionArgumentCaptor.getValue().execute();
74  
75          verifyStatic();
76          Keys.getOrAdd(ChangePasswordConfig.class.getCanonicalName());
77  
78          verifyStatic();
79          Keys.getOrAdd(ChangePasswordActor.class.getCanonicalName());
80  
81          ArgumentCaptor<ApplyFunctionToArgumentsStrategy> argumentCaptor = ArgumentCaptor.forClass(ApplyFunctionToArgumentsStrategy.class);
82  
83          verifyStatic();
84          IOC.register(eq(changePasswordActorKey), argumentCaptor.capture());
85  
86          IObject configObj = mock(IObject.class);
87          IKey configKey = mock(IKey.class);
88          when(Keys.getOrAdd(ChangePasswordConfig.class.getCanonicalName())).thenReturn(configKey);
89          ChangePasswordConfig config = mock(ChangePasswordConfig.class);
90          when(IOC.resolve(configKey, configObj)).thenReturn(config);
91  
92          ChangePasswordActor actor = mock(ChangePasswordActor.class);
93          whenNew(ChangePasswordActor.class).withArguments(config).thenReturn(actor);
94  
95          argumentCaptor.getValue().resolve(configObj);
96  
97          verifyNew(ChangePasswordActor.class);
98      }
