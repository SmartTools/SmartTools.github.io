3   import info.smart_tools.smartactors.core.db_storage.interfaces.CompiledQuery;
4   import info.smart_tools.smartactors.core.db_task.delete.wrappers.DeletionQuery;

6   import info.smart_tools.smartactors.core.itask.exception.TaskExecutionException;
7   import info.smart_tools.smartactors.core.sql_commons.JDBCCompiledQuery;
8   
9   import javax.annotation.Nonnull;
10  import java.sql.SQLException;
11  
12  /**
13   * Common deletion task executor.
14   */
15  public abstract class DBDeleteTask implements IDatabaseTask {
16  
17      /**
18       * Default constructor.
19       */
20      protected DBDeleteTask() {}
21  
22      /**
23       * Executes deletion query of rows to database.
24       *
25       * @param query - a compiled executable query to database.
26       * @param message - source message with parameters for query.
27       *
28       * @throws TaskExecutionException when number of deleted rows not equals of number of rows which had to be removed.
29       */
30      protected void execute(final @Nonnull CompiledQuery query, final @Nonnull DeletionQuery message)
31              throws TaskExecutionException {
32          try {
33              int nDeleted = ((JDBCCompiledQuery) query).getPreparedStatement().executeUpdate();
34  
35              if (nDeleted != message.countDocumentIds()) {
36                  throw new TaskExecutionException("Delete query failed: wrong count of documents is deleted.");
37              }
38          } catch (SQLException e) {
39              throw new TaskExecutionException("Deletion query execution failed because of SQL exception.", e);
40          }
41      }
42  }
