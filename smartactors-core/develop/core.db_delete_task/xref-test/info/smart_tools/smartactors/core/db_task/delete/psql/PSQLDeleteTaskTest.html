54      @Test
55      public void prepareDeletionQueryTest() throws Exception {
56          StorageConnection connection = mock(StorageConnection.class);
57          DBDeleteTask deleteTask = PSQLDeleteTask.create();
58          deleteTask.setConnection(connection);
59  
60          IKey wrapperKey = mock(IKey.class);
61          IKey queryKey = mock(IKey.class);
62          when(Keys.getOrAdd(DeletionQuery.class.toString())).thenReturn(wrapperKey);
63          when(Keys.getOrAdd(CompiledQuery.class.toString())).thenReturn(queryKey);
64          when(IOC.resolve(eq(wrapperKey), anyObject())).thenReturn(queryMessage);
65          when(IOC.resolve(eq(queryKey), eq(connection), eq(PSQLDeleteTask.class.toString()), anyObject()))
66                  .thenReturn(compiledQuery);
67  
68          IObject deletionQuery = mock(IObject.class);
69          deleteTask.prepare(deletionQuery);
70  
71          verify(queryMessage, times(2)).countDocumentIds();
72          verifyStatic(times(1));
73          IOC.resolve(eq(wrapperKey), eq(deletionQuery));
74  
75          Field queryField  = deleteTask.getClass().getDeclaredField("query");
76          queryField.setAccessible(true);
77          assertEquals(queryField.get(deleteTask), compiledQuery);
78  
79          Field messageField = deleteTask.getClass().getDeclaredField("message");
80          messageField.setAccessible(true);
81          assertEquals(messageField.get(deleteTask), queryMessage);
82      }
