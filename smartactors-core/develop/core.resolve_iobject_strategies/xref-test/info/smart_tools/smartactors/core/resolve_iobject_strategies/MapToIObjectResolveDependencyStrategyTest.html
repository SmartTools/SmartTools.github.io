28  @RunWith(PowerMockRunner.class)
29  @PrepareForTest({IOC.class, Keys.class})
30  public class MapToIObjectResolveDependencyStrategyTest {
31  
32      private MapToIObjectResolveDependencyStrategy strategy;
33  
34      @Before
35      public void setUp() throws Exception {
36  
37          mockStatic(IOC.class);
38          mockStatic(Keys.class);
39  
40          strategy = new MapToIObjectResolveDependencyStrategy();
41      }
42  
43      @Test
44      public void ShouldCorrectConvertMapToIObject() throws Exception {
45  
46          IKey key = mock(IKey.class);
47          when(Keys.getOrAdd(IFieldName.class.getCanonicalName())).thenReturn(key);
48  
49          IFieldName fieldName1 = mock(IFieldName.class);
50          IFieldName fieldName2 = mock(IFieldName.class);
51          when(IOC.resolve(key, "key1")).thenReturn(fieldName1);
52          when(IOC.resolve(key, "key2")).thenReturn(fieldName2);
53  
54          Map<String, Object> map = new LinkedHashMap<>();
55          map.put("key1", 123);
56          map.put("key2", "value");
57  
58          IObject result = strategy.resolve(map);
59          assertEquals(result.getValue(fieldName1), 123);
60          assertEquals(result.getValue(fieldName2), "value");
61      }
62  
63      @Test(expected = ResolveDependencyStrategyException.class)
64      public void ShouldThrowException_When_AnyErrorIsOccurred() throws Exception {
65  
66          when(IOC.resolve(any(IKey.class), anyString())).thenThrow(new ResolutionException(""));
67  
68          Map<String, Object> map = new HashMap<>();
69          map.put("key", "value");
70          strategy.resolve(map);
71          fail();
72      }
73  
74      @Test(expected = ResolveDependencyStrategyException.class)
75      public void ShouldThrowException_When_NullIsPassed() throws Exception {
76  
77          strategy.resolve(null);
78          fail();
79      }
