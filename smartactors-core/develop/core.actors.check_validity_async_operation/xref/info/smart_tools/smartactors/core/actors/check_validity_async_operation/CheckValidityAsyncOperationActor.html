10  /**
11   * The Actor for comparing of two identifiers of asynchronous operations:
12   * one came from the client, and the list of remaining is stored in session.
13   * If there is no coincidence, then throw new CheckException
14   */
15  public class CheckValidityAsyncOperationActor {
16  
17      /**
18       * Constructor for CheckValidityAsyncOperationActor
19       * @param config contains any params
20       */
21      public CheckValidityAsyncOperationActor(final IObject config) {
22      }
23  
24      /**
25       * Check validity asynchronous operation, if invalid - throw exception
26       * @param message input message
27       * @throws InvalidAsyncOperationIdException
28       */
29      public void check(final CheckValidityMessage message) throws InvalidAsyncOperationIdException {
30          try {
31              List<String> validIds = message.getIdentifiers();
32              if (validIds == null || validIds.isEmpty()) {
33                  throw new ReadValueException("List of valid id is empty");
34              }
35              if (!validIds.contains(message.getAsyncOperationId())) {
36                  throw new InvalidAsyncOperationIdException("List with all identifiers not contains Id from client.");
37              }
38          } catch (ReadValueException e) {
39              throw new InvalidAsyncOperationIdException(e);
40          }
41  
42      }
43  
44  }
