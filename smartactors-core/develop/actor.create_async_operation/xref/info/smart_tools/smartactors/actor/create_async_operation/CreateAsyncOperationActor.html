20  public class CreateAsyncOperationActor {
21  
22      /**
23       * Constructor needed for registry actor
24       * @param params iobject
25       */
26      public CreateAsyncOperationActor(final IObject params) {
27      }
28  
29      /**
30       * Generates token and creates asynchronous operation.
31       * Sets token to session and response.
32       * @param message {
33       *                "sessionId": "session identifier for save as async data",
34       *                "expiredTime": "TTL for async operation"
35       * }
36       * @throws CreateAsyncOperationActorException for creation error
37       */
38      public void create(final CreateAsyncOperationMessage message) throws CreateAsyncOperationActorException {
39  
40          try {
41              //TODO:: move generate to util class and add server number
42              String token = String.valueOf(UUID.randomUUID());
43              IAsyncOperationCollection collection = IOC.resolve(Keys.getOrAdd(IAsyncOperationCollection.class.toString()));
44              String expiredTime = message.getExpiredTime();
45              AuthOperationData authOperationData = IOC.resolve(Keys.getOrAdd(AuthOperationData.class.toString()));
46              authOperationData.setSessionId(message.getSessionId());
47  //            collection.createAsyncOperation(((IObjectWrapper) authOperationData).getIObjects()[0], token, expiredTime);
48              collection.createAsyncOperation(IOC.resolve(Keys.getOrAdd(IObject.class.toString()), authOperationData), token, expiredTime);
49  
50              //NOTE: this setter should set token to session and to response!
51              message.setAsyncOperationToken(token);
52          } catch (ReadValueException | ResolutionException | ChangeValueException | CreateAsyncOperationException e) {
53              throw new CreateAsyncOperationActorException("Can't create async operation.", e);
54          }
55      }
56  }
