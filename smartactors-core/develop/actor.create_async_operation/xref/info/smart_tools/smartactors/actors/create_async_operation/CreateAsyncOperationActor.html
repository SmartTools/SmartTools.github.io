22  /**
23   * Actor for creation asynchronous operation
24   */
25  public class CreateAsyncOperationActor {
26  
27      private IAsyncOperationCollection collection;
28  
29      //TODO:: this format should be setted for whole project?
30      private static final DateTimeFormatter FORMATTER = DateTimeFormatter.ofPattern("MM-dd-yyyy HH:mm:ss");
31  
32      /**
33       * Constructor needed for registry actor
34       * @param params iobject
35       * @throws CreateAsyncOperationActorException if any error is occurred
36       */
37      public CreateAsyncOperationActor(final IObject params) throws CreateAsyncOperationActorException {
38          try {
39              IField collectionNameField = IOC.resolve(Keys.getOrAdd(IField.class.getCanonicalName()), "collectionName");
40              collection = IOC.resolve(
41                  Keys.getOrAdd(IAsyncOperationCollection.class.getCanonicalName()), (String) collectionNameField.in(params)
42              );
43          } catch (ReadValueException | InvalidArgumentException e) {
44              throw new CreateAsyncOperationActorException("Can't read collection name from message", e);
45          } catch (ResolutionException e) {
46              throw new CreateAsyncOperationActorException("Can't get key or resolve dependency", e);
47          }
48      }
49  
50      /**
51       * Generates token and creates asynchronous operation.
52       * Sets token to session and response.
53       * @param message {
54       *                "sessionId": "session identifier for save as async data",
55       *                "expiredTime": "TTL for async operation"
56       * }
57       * @throws CreateAsyncOperationActorException for creation error
58       */
59      public void create(final CreateAsyncOperationMessage message) throws CreateAsyncOperationActorException {
60  
61          try {
62              //TODO:: move generate to util class and add server number
63              String token = String.valueOf(UUID.randomUUID());
64              Integer amountOfHoursToExpireFromNow = message.getExpiredTime();
65              String expiredTime = LocalDateTime.now().plusHours(amountOfHoursToExpireFromNow).format(FORMATTER);
66              IObject authOperationData = message.getOperationData();
67              collection.createAsyncOperation(authOperationData, token, expiredTime);
68  
69              message.setAsyncOperationToken(token);
70  
71              List<String> availableTokens = message.getOperationTokens();
72              if (availableTokens == null) {
73                  message.setOperationTokens(Collections.singletonList(token));
74              } else {
75                  availableTokens.add(token);
76                  message.setOperationTokens(availableTokens);
77              }
78          } catch (ReadValueException | ChangeValueException | CreateAsyncOperationException e) {
79              throw new CreateAsyncOperationActorException("Can't create async operation.", e);
80          }
81      }
82  }
