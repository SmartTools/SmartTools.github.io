3   import info.smart_tools.smartactors.core.iobject.IObject;
4   
5   /**
6    * Chain of message receivers.
7    */
8   public interface IReceiverChain {
9       /**
10       * Get {@link IMessageReceiver} at given index.
11       *
12       * @param index    index of the receiver in the chain
13       * @return receiver at given index or {@code null} if there is no such receiver
14       */
15      IMessageReceiver get(int index);
16  
17      /**
18       * Get arguments object that should be passed to the receiver at given index of this chain.
19       *
20       * @param index    index of the receiver in the chain
21       * @return arguments that should be passed to the receiver
22       * @see IMessageReceiver#receive(IMessageProcessor)
23       */
24      IObject getArguments(int index);
25  
26      /**
27       * Get name of this chain.
28       *
29       * @return name of this chain
30       */
31      String getName();
32  
33      /**
34       * Get the chain of receivers (and environments) that has to be executed if exception occurs during execution of this chain.
35       *
36       * @param exception the exception occurred
37       * @return the chain of receivers (and environments) that has to be executed if exception occurs during execution of this chain or
38       *         {@code null} if no such chain defined for this one
39       */
40      IObject getExceptionalChainAndEnvironments(Throwable exception);
41  }
