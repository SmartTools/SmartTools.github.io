1   package info.smart_tools.smartactors.core.message_processing;
2   
3   import info.smart_tools.smartactors.core.iioccontainer.exception.ResolutionException;
4   import info.smart_tools.smartactors.core.invalid_argument_exception.InvalidArgumentException;
5   import info.smart_tools.smartactors.core.iobject.IObject;
6   import info.smart_tools.smartactors.core.iobject.exception.ChangeValueException;
7   import info.smart_tools.smartactors.core.message_processing.exceptions.AsynchronousOperationException;
8   
9   /**
10   * Interface for a object responding for processing of a single message.
11   */
12  public interface IMessageProcessor {
13      /**
14       * Returns the message object.
15       *
16       * @return the message being processed
17       */
18      IObject getMessage();
19  
20      /**
21       * Returns the context object of the message being processed. Context contains all the non-serializable data related
22       * to the message.
23       *
24       * @return the context of message processing
25       */
26      IObject getContext();
27  
28      /**
29       * Returns the response object being formed for the processed message. The response object will be sent to the
30       * sender of the message being processed (and may be serialized in process of sending) so should contain only
31       * serializable data.
32       *
33       * @return the response for the message
34       */
35      IObject getResponse();
36  
37      /**
38       * Returns the {@link IMessageProcessingSequence} managing order of receivers receiving this message.
39       *
40       * @return the sequence of receivers that should receive the message
41       */
42      IMessageProcessingSequence getSequence();
43  
44      /**
45       * Returns the message environment object. The environment object should contain at least the following fields:
46       * - "message" - the message itself (as instance of {@link IObject})
47       * - "context" - the {@link IObject} containing non-serializable objects related to processing of the message
48       * - "response" - the {@link IObject} containing fields that will be sent as a response to the processed message
49       * - "sequence" - the {@link IMessageProcessingSequence} instance associated with the message
50       * - "arguments" - the arguments {@link IObject} passed from {@link IReceiverChain}
51       *
52       * @return the environment object
53       */
54      IObject getEnvironment();
55  
56      /**
57       * Set a global configuration object to use for processing of next message.
58       *
59       * @param config    the configuration object
60       * @throws InvalidArgumentException if {@code config} is {@code null}
61       */
62      void setConfig(IObject config) throws InvalidArgumentException;
63  
64      /**
65       * Start processing new message using this instance of {@link IMessageProcessor}.
66       *
67       * @param message    the message to process
68       * @param context    the context of message processing
69       * @throws InvalidArgumentException if message is {@code null}
70       * @throws InvalidArgumentException if context is {@code null}
71       * @throws ResolutionException if fails to resolve any dependency
72       * @throws ChangeValueException if modification of internal environment object fails
73       */
74      void process(IObject message, IObject context)
75              throws InvalidArgumentException, ResolutionException, ChangeValueException;
76  
77      /**
78       * Notify this message processor that currently called message receiver started a asynchronous operation and the
79       * processor should not call any following receivers until it is completed.
80       *
81       * @throws AsynchronousOperationException if called outside of {@link IMessageReceiver#receive} call after completion
82       *                                             of all asynchronous operations
83       */
84      void pauseProcess() throws AsynchronousOperationException;
85  
86      /**
87       * Notify this message processor that a asynchronous operation (on start of which the message processor was notified
88       * by call of {@link #pauseProcess()}) is completed.
89       *
90       * @param e    the exception occurred during asynchronous operation or {@code null} if operation completed successful
91       * @throws AsynchronousOperationException if all asynchronous operations are already completed i.e. number of
92       *                                             calls to #continueProcess is greater than number of calls to {@link
93       *                                             #pauseProcess()} since start of execution of last receiver
94       */
95      void continueProcess(Throwable e) throws AsynchronousOperationException;
96  }
