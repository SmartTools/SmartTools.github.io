4   import info.smart_tools.smartactors.core.istrategy_container.IStrategyContainer;

7   import java.util.Map;
8   import java.util.concurrent.ConcurrentHashMap;
9   
10  /**
11   * Implementation of {@link info.smart_tools.smartactors.core.istrategy_container.IStrategyContainer}
12   * <pre>
13   * Simple key-value storage
14   *  - key is a unique object identifier
15   *  - value is a instance of {@link info.smart_tools.smartactors.core.iresolve_dependency_strategy.IResolveDependencyStrategy}
16   * </pre>
17   */
18  public class StrategyContainer implements IStrategyContainer {
19  
20      /**
21       * Local storage
22       */
23      private Map<Object, IResolveDependencyStrategy> strategyStorage = new ConcurrentHashMap<Object, IResolveDependencyStrategy>();
24  
25      /**
26       * Resolve {@link IResolveDependencyStrategy} by given unique object identifier
27       * @param key unique object identifier
28       * @return instance of {@link IResolveDependencyStrategy}
29       * @throws StrategyContainerException if any errors occurred
30       */
31      public IResolveDependencyStrategy resolve(final Object key)
32              throws StrategyContainerException {
33          return strategyStorage.get(key);
34      }
35  
36      /**
37       * Register new dependency of {@link IResolveDependencyStrategy} instance by unique object identifier
38       * @param key unique object identifier
39       * @param strategy instance of {@link IResolveDependencyStrategy}
40       * @throws StrategyContainerException if any error occurred
41       */
42      public void register(final Object key, final IResolveDependencyStrategy strategy)
43              throws StrategyContainerException {
44          strategyStorage.put(key, strategy);
45      }
46  
47      /**
48       * Remove existing dependency of {@link IResolveDependencyStrategy} by unique object identifier
49       * @param key unique object identifier
50       * @throws StrategyContainerException if any error occurred
51       */
52      public void remove(final Object key)
53              throws StrategyContainerException {
54          strategyStorage.remove(key);
55      }
56  }
