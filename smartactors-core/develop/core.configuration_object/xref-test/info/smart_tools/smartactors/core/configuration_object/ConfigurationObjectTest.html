243     @Test
244     public void checkCreationAndResolutionSomeFields()
245             throws Exception {
246         ConfigurationObject co = new ConfigurationObject(this.configString);
247 
248         List<IObject> in_getIntValue = (List<IObject>) ((IObject)co.getValue(new FieldName("wrapper"))).getValue(new FieldName("in_getIntValue"));
249         assertEquals(in_getIntValue.get(0).getValue(new FieldName("name")), "wds_getter_strategy");
250         assertEquals(((List)in_getIntValue.get(0).getValue(new FieldName("args"))).get(0), "message/IntValue");
251 
252         List<List<IObject>> out_setIntValue = (List<List<IObject>>) ((IObject)co.getValue(new FieldName("wrapper"))).getValue(new FieldName("out_setIntValue"));
253         assertEquals(out_setIntValue.get(0).get(0).getValue(new FieldName("name")), "wds_target_strategy");
254         assertEquals(((List)out_setIntValue.get(0).get(0).getValue(new FieldName("args"))).get(0), "local/value");
255         assertEquals(((List)out_setIntValue.get(0).get(0).getValue(new FieldName("args"))).get(1), "response/IntValue");
256 
257         List<List<IObject>> out_setIObject = (List<List<IObject>>) ((IObject)co.getValue(new FieldName("wrapper"))).getValue(new FieldName("out_setIObject"));
258         assertEquals(out_setIObject.get(0).get(0).getValue(new FieldName("name")), "wds_target_strategy");
259         assertEquals(((List)out_setIObject.get(0).get(0).getValue(new FieldName("args"))).get(0), "local/value");
260         assertEquals(((List)out_setIObject.get(0).get(0).getValue(new FieldName("args"))).get(1), "response/IObject");
261 
262         List<IObject> in_getStringValue = (List<IObject>) ((IObject)co.getValue(new FieldName("wrapper"))).getValue(new FieldName("in_getStringValue"));
263         assertEquals(in_getStringValue.get(0).getValue(new FieldName("name")), "wds_getter_strategy");
264         assertEquals(((List)in_getStringValue.get(0).getValue(new FieldName("args"))).get(0), "message/StringValue");
265     }
