18  /**
19   * Task-facade for read task for async operations collection
20   */
21  public class GetAsyncOperationTask implements IDatabaseTask {
22  
23      private IDatabaseTask getItemTask;
24  
25      /**
26       * Constructor
27       * @param getItemTask nested task for read object
28       */
29      public GetAsyncOperationTask(final IDatabaseTask getItemTask) {
30          this.getItemTask = getItemTask;
31      }
32  
33      /**
34       * @param query query object
35       *              {
36       *                  "collectionName": "name of async operation collection",
37       *                  "token": "unique token of async operation for search"
38       *              }
39       * @throws TaskPrepareException for prepare error
40       */
41      @Override
42      public void prepare(final IObject query) throws TaskPrepareException {
43          try {
44              GetAsyncOperationQuery srcQueryObject = IOC.resolve(Keys.getOrAdd(GetAsyncOperationQuery.class.toString()), query);
45              AsyncOperationTaskQuery criteriaQuery = IOC.resolve(Keys.getOrAdd(AsyncOperationTaskQuery.class.toString()));
46  
47              EQMessage tokenCondition = IOC.resolve(Keys.getOrAdd(EQMessage.class.toString()));
48              tokenCondition.setEq(srcQueryObject.getToken());
49              criteriaQuery.setToken(tokenCondition);
50  
51              srcQueryObject.setPageNumber(1);
52              srcQueryObject.setPageSize(1);
53              srcQueryObject.setQuery(criteriaQuery);
54  
55              getItemTask.prepare(query);
56          } catch (ResolutionException e) {
57              throw new TaskPrepareException("Can't create ISearchQuery from input query", e);
58          } catch (ChangeValueException | ReadValueException e) {
59              throw new TaskPrepareException("Can't change value in one of IObjects", e);
60          }
61      }
62  
63      @Override
64      public void setConnection(final StorageConnection connection) throws TaskSetConnectionException {
65          getItemTask.setConnection(connection);
66      }
67  
68      @Override
69      public void execute() throws TaskExecutionException {
70          getItemTask.execute();
71      }
72  }
