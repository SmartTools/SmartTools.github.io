45  public class AsyncOperationCollection implements IAsyncOperationCollection {

47      private IPool connectionPool;
48      private CollectionName collectionName;

51      /**
52       * Constructor for implementation
53       * @param connectionPool connection pool
54       * @param collectionName string with name
55       * @throws InvalidArgumentException if we can't create collection name or field
56       */
57      public AsyncOperationCollection(final IPool connectionPool, final String collectionName) throws InvalidArgumentException {
58          this.connectionPool = connectionPool;
59          try {
60              this.idField = IOC.resolve(Keys.getOrAdd(IField.class.toString()), "id");
61              this.collectionName = CollectionName.fromString(collectionName);
62          } catch (QueryBuildException e) {
63              throw new InvalidArgumentException("Can't create async operations collection.", e);
64          } catch (ResolutionException e) {
65              throw new InvalidArgumentException("Can't create field", e);
66          }
67      }

69      @Override
70      public IObject getAsyncOperation(final String token) throws GetAsyncOperationException {
71  
72          try (IPoolGuard poolGuard = new PoolGuard(connectionPool)) {
73              IDatabaseTask getItemTask = IOC.resolve(Keys.getOrAdd(GetAsyncOperationTask.class.toString()));
74              if (getItemTask == null) {
75                  IDatabaseTask nestedTask = IOC.resolve(
76                      Keys.getOrAdd(IDatabaseTask.class.toString()), GetAsyncOperationTask.class.toString()
77                  );
78                  if (nestedTask == null) {
79                      throw new GetAsyncOperationException("Can't create nested task for getItem task.");
80                  }
81                  getItemTask = new GetAsyncOperationTask(nestedTask);
82                  IOC.register(Keys.getOrAdd(GetAsyncOperationTask.class.toString()), new SingletonStrategy(getItemTask));
83              }
84              GetAsyncOperationQuery getItemQuery = IOC.resolve(Keys.getOrAdd(GetAsyncOperationQuery.class.toString()));
85              getItemQuery.setCollectionName(collectionName);
86              getItemQuery.setToken(token);
87              getItemTask.setConnection(IOC.resolve(Keys.getOrAdd(StorageConnection.class.toString()), poolGuard.getObject()));
88              getItemTask.prepare(getItemQuery.wrapped());
89              getItemTask.execute();
90              List<IObject> searchResult = getItemQuery.getSearchResult();
91              if (searchResult == null || searchResult.isEmpty()) {
92                  throw new GetAsyncOperationException("Can't find operation.");
93              }
94  
95              return searchResult.get(0);
96          } catch (ResolutionException e) {
97              throw new GetAsyncOperationException("Can't resolve object during get operation.", e);
98          } catch (InvalidArgumentException | RegistrationException e) {
99              throw new GetAsyncOperationException("Can't register strategy for getItem task.", e);
100         } catch (PoolGuardException e) {
101             throw new GetAsyncOperationException("Can't get connection from pool.", e);
102         } catch (ReadValueException | ChangeValueException e) {
103             throw new GetAsyncOperationException("Can't read asynchronous operation.", e);
104         } catch (TaskSetConnectionException e) {
105             throw new GetAsyncOperationException("Can't set connection to read task.", e);
106         } catch (TaskPrepareException e) {
107             throw new GetAsyncOperationException("Error during preparing read task.", e);
108         } catch (TaskExecutionException e) {
109             throw new GetAsyncOperationException("Error during execution read task.", e);
110         }
111     }

113     @Override
114     public void createAsyncOperation(final IObject data, final String token, final String expiredTime)
115         throws CreateAsyncOperationException {
116 
117         try (IPoolGuard poolGuard = new PoolGuard(connectionPool)) {
118             IDatabaseTask task = IOC.resolve(Keys.getOrAdd(CreateAsyncOperationTask.class.toString()));
119             if (task == null) {
120                 IDatabaseTask nestedTask = IOC.resolve(
121                         Keys.getOrAdd(IDatabaseTask.class.toString()), CreateAsyncOperationTask.class.toString()
122                 );
123                 if (nestedTask == null) {
124                     throw new CreateAsyncOperationException("Can't create nested task for createtItem task.");
125                 }
126                 task = new CreateAsyncOperationTask(nestedTask);
127                 IOC.register(Keys.getOrAdd(GetAsyncOperationTask.class.toString()), new SingletonStrategy(task));
128             }
129             CreateOperationQuery query = IOC.resolve(Keys.getOrAdd(CreateOperationQuery.class.toString()));
130             query.setCollectionName(collectionName);
131             query.setAsyncData(data);
132             query.setExpiredTime(expiredTime);
133             query.setToken(token);
134             task.setConnection((StorageConnection) poolGuard.getObject());
135             task.prepare(query.getIObject());
136             task.execute();
137         } catch (Exception e) {
138             throw new CreateAsyncOperationException("Failed to create async operation.");
139         }
140     }

142     @Override
143     public void complete(final IObject asyncOperation) throws CompleteAsyncOperationException {
144 
145         try (IPoolGuard poolGuard = new PoolGuard(connectionPool)) {
146             IDatabaseTask updateTask = IOC.resolve(Keys.getOrAdd(UpdateAsyncOperationTask.class.toString()));
147             if (updateTask == null) {
148                 IDatabaseTask nestedTask = IOC.resolve(
149                     Keys.getOrAdd(IDatabaseTask.class.toString()), UpdateAsyncOperationTask.class.toString()
150                 );
151                 if (nestedTask == null) {
152                     throw new CompleteAsyncOperationException("Can't create nested task for update task.");
153                 }
154                 updateTask = new UpdateAsyncOperationTask(nestedTask);
155                 IOC.register(Keys.getOrAdd(UpdateAsyncOperationTask.class.toString()), new SingletonStrategy(updateTask));
156             }
157             UpdateAsyncOperationQuery upsertQuery = IOC.resolve(Keys.getOrAdd(UpdateAsyncOperationQuery.class.toString()));
158             upsertQuery.setCollectionName(collectionName);
159             upsertQuery.setUpdateItem(asyncOperation);
160 
161             updateTask.setConnection(IOC.resolve(Keys.getOrAdd(StorageConnection.class.toString()), poolGuard.getObject()));
162             updateTask.prepare(IOC.resolve(Keys.getOrAdd(IObject.class.toString()), upsertQuery));
163             updateTask.execute();
164         } catch (TaskExecutionException e) {
165             throw new CompleteAsyncOperationException("Error during execution complete.", e);
166         } catch (PoolGuardException e) {
167             throw new CompleteAsyncOperationException("Can't get connection from pool.", e);
168         } catch (TaskSetConnectionException e) {
169             throw new CompleteAsyncOperationException("Can't set connection to update task.", e);
170         } catch (TaskPrepareException e) {
171             throw new CompleteAsyncOperationException("Error during preparing update task.", e);
172         } catch (InvalidArgumentException | RegistrationException e) {
173             throw new CompleteAsyncOperationException("Can't register strategy for update task.", e);
174         } catch (ChangeValueException e) {
175             throw new CompleteAsyncOperationException("Can't complete async operation.", e);
176         } catch (ResolutionException e) {
177             throw new CompleteAsyncOperationException("Can't resolve async operation object.", e);
178         } catch (UpdateAsyncOperationException e) {
179             throw new CompleteAsyncOperationException("Can't create new UpdateAsyncOperationTask.", e);
180         }
181     }

183     @Override
184     public void delete(final String token) throws DeleteAsyncOperationException {
185 
186         try (IPoolGuard poolGuard = new PoolGuard(connectionPool)) {
187             IDatabaseTask deleteTask = IOC.resolve(Keys.getOrAdd(DeleteAsyncOperationTask.class.toString()));
188             if (deleteTask == null) {
189                 IDatabaseTask nestedTask = IOC.resolve(
190                     Keys.getOrAdd(IDatabaseTask.class.toString()), DeleteAsyncOperationTask.class.toString()
191                 );
192                 if (nestedTask == null) {
193                     throw new DeleteAsyncOperationException("Can't create nested task for update task.");
194                 }
195                 deleteTask = new DeleteAsyncOperationTask(nestedTask);
196                 IOC.register(Keys.getOrAdd(DeleteAsyncOperationTask.class.toString()), new SingletonStrategy(deleteTask));
197             }
198             DeleteAsyncOperationQuery deleteQuery = IOC.resolve(Keys.getOrAdd(DeleteAsyncOperationQuery.class.toString()));
199             deleteQuery.setCollectionName(collectionName);
200             IObject deleteItem = getAsyncOperation(token);
201             deleteQuery.setDocumentIds(Collections.singletonList(idField.in(deleteItem)));
202 
203             deleteTask.setConnection(IOC.resolve(Keys.getOrAdd(StorageConnection.class.toString()), poolGuard.getObject()));
204             deleteTask.prepare(IOC.resolve(Keys.getOrAdd(IObject.class.toString()), deleteQuery));
205             deleteTask.execute();
206         } catch (TaskExecutionException e) {
207             throw new DeleteAsyncOperationException("Error during execution complete.", e);
208         } catch (PoolGuardException e) {
209             throw new DeleteAsyncOperationException("Can't get connection from pool.", e);
210         } catch (TaskSetConnectionException e) {
211             throw new DeleteAsyncOperationException("Can't set connection to update task.", e);
212         } catch (TaskPrepareException e) {
213             throw new DeleteAsyncOperationException("Error during preparing update task.", e);
214         } catch (InvalidArgumentException | RegistrationException e) {
215             throw new DeleteAsyncOperationException("Can't register strategy for update task.", e);
216         } catch (ReadValueException | ChangeValueException e) {
217             throw new DeleteAsyncOperationException("Can't complete async operation.", e);
218         } catch (ResolutionException e) {
219             throw new DeleteAsyncOperationException("Can't resolve async operation object.", e);
220         } catch (GetAsyncOperationException e) {
221             throw new DeleteAsyncOperationException("Can't get async operation by token.", e);
222         }
223     }
