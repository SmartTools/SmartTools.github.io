12  public class Base64Test {
13      private final ICodec base64 = Base64.create();
14      private final byte[] SOURCE_MSG = ("Hello my young friend. " +
15              "You can see how it's is a nice text will converted to some shit, " +
16              "but to see it, you should debug this text. Here are the things here.").getBytes();
17  
18      private final byte[] BASE64_MSG = ("SGVsbG8gbXkgeW91bmcgZnJpZW5kLi" +
19              "BZb3UgY2FuIHNlZSBob3cgaXQncyBpcyBhIG5pY2UgdGV4" +
20              "dCB3aWxsIGNvbnZlcnRlZCB0byBzb21lIHNoaXQsIGJ1dC" +
21              "B0byBzZWUgaXQsIHlvdSBzaG91bGQgZGVidWcgdGhpcyB0" +
22              "ZXh0LiBIZXJlIGFyZSB0aGUgdGhpbmdzIGhlcmUu").getBytes();
23  
24      @Test
25      public void encodeBase64Test() throws EncodingException {
26          byte[] base64EncodeResult = base64.encode(SOURCE_MSG);
27          assertTrue(Arrays.equals(base64EncodeResult, BASE64_MSG));
28      }
29  
30      @Test
31      public void decodeBase64Test() throws DecodingException {
32          byte[] base64DecodeResult = base64.decode(BASE64_MSG);
33          assertTrue(Arrays.equals(base64DecodeResult, SOURCE_MSG));
34      }
