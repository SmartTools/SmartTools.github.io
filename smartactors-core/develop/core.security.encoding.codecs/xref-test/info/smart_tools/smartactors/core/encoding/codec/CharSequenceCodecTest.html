16  public class CharSequenceCodecTest {
17      private ICharSequenceCodec charSequenceCodec;
18  
19      private final String SOURCE_MSG = "Hello my young friend. " +
20              "You can see how it's is a nice text will converted to some shit, " +
21              "but to see it, you should debug this text. Here are the things here.";
22  
23      // java utf-8 converter.
24      private byte[] UTF_8_MSG;
25  
26      @Before
27      public void setUp() throws UnsupportedEncodingException, InvalidArgumentException {
28          charSequenceCodec = CharSequenceCodec.create("utf-8");
29          UTF_8_MSG = SOURCE_MSG.getBytes("UTF-8");
30      }
31  
32      @Test
33      public void encodeHexTest() throws EncodingException {
34          byte[] utf8EncodeResult = charSequenceCodec.encode(SOURCE_MSG);
35          assertTrue(Arrays.equals(utf8EncodeResult, UTF_8_MSG));
36      }
37  
38      @Test
39      public void decodeHexTest() throws DecodingException {
40          String utf8DecodeResult = charSequenceCodec.decode(UTF_8_MSG);
41          assertTrue(utf8DecodeResult.equals(SOURCE_MSG));
42      }
43  
44      @Test
45      public void ShouldBeEquals_When_EncodeAndDecode() throws EncodingException, DecodingException {
46          assertEquals(charSequenceCodec.decode(charSequenceCodec.encode(SOURCE_MSG)), SOURCE_MSG);
47      }
48  }
