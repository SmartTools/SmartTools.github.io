14  /**
15   * Actor for read asynchronous operation by token
16   */
17  public class GetAsyncOperationActor {
18  
19      /**
20       * Constructor needed for registry actor
21       * @param params iobject
22       */
23      public GetAsyncOperationActor(final IObject params) {
24      }
25  
26      /**
27       * Gets operation from collection of asynchronous operations.
28       * Sets operation to message.
29       * @param message {
30       *                "token": "unique identifier of operation"
31       * }
32       * @throws GetAsyncOperationActorException for read error
33       */
34      public void getOperation(final GetAsyncOperationMessage message) throws GetAsyncOperationActorException {
35  
36          try {
37              IAsyncOperationCollection collection = IOC.resolve(Keys.getOrAdd(IAsyncOperationCollection.class.toString()));
38              String token = message.getToken();
39              IObject asyncOperation = collection.getAsyncOperation(token);
40              if (asyncOperation == null) {
41                  throw new GetAsyncOperationActorException("Asynchronous operation by token " + token + " is null");
42              }
43              message.setAsyncOperation(asyncOperation);
44          } catch (ResolutionException | ChangeValueException | GetAsyncOperationException | ReadValueException e) {
45              throw new GetAsyncOperationActorException("Can't get async operation by token.", e);
46          }
47      }
48  }
