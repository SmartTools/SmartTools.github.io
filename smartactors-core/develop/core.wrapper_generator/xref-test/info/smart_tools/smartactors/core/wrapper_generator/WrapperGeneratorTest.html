308     @Test
309     public void checkCreationAndUsageWrapperByInterface()
310             throws Exception {
311         IWrapperGenerator wg = new WrapperGenerator(null);
312         IObject binding = getBinding();
313 
314         IWrapper inst = wg.generate(IWrapper.class, binding);
315 
316         assertNotNull(inst);
317 
318         IObject message = mock(IObject.class);
319         IObject context = mock(IObject.class);
320         IObject response = mock(IObject.class);
321 
322         when(message.getValue(new FieldName("IntValue"))).thenReturn(1);
323         when(message.getValue(new FieldName("StringValue"))).thenReturn("abc");
324         TestClass testClass = new TestClass();
325         when(message.getValue(new FieldName("TestClassValue"))).thenReturn(testClass);
326         List<Integer> listOfInt = new ArrayList<Integer>(){{add(1);}};
327         when(message.getValue(new FieldName("ListOfInt"))).thenReturn(listOfInt);
328         List<String> listOfString = new ArrayList<String>(){{add("abc");}};
329         when(message.getValue(new FieldName("ListOfString"))).thenReturn(listOfString);
330         List<TestClass> listOfTestClasses = new ArrayList<TestClass>(){{new TestClass();}};
331         when(message.getValue(new FieldName("ListOfTestClasses"))).thenReturn(listOfTestClasses);
332         when(message.getValue(new FieldName("BoolValue"))).thenReturn(true);
333         IInnerWrapper innerWrapper = mock(IInnerWrapper.class);
334         when(message.getValue(new FieldName("WrappedIObject"))).thenReturn(innerWrapper);
335         IObject iObject = mock(IObject.class);
336         when(message.getValue(new FieldName("IObject"))).thenReturn(iObject);
337         when(context.getValue(new FieldName("StringIInnerMap"))).thenReturn(new HashMap<String, IInnerWrapper>(){{put("abc", innerWrapper);}});
338 
339         ((IObjectWrapper)inst).init(message, context, response);
340 
341         int intResult = inst.getIntValue();
342         String stringResult = inst.getStringValue();
343         TestClass testClassResult = inst.getTestClassValue();
344         List<Integer> integerList = inst.getListOfInt();
345         List<String> stringList = inst.getListOfString();
346         List<TestClass> testClasses = inst.getListOfTestClasses();
347         boolean boolResult = inst.getBoolValue();
348         IInnerWrapper innerWrapperResult = inst.getWrappedIObject();
349         IObject iObjectResult = inst.getIObject();
350         Map<String, IInnerWrapper> mapResult = inst.getStringIInnerMap();
351         IObject[] iObjects = ((IObjectWrapper) inst).getIObjects();
352 
353         assertEquals(intResult, 1);
354         assertEquals(stringResult, "abc");
355         assertEquals(testClassResult, testClass);
356         assertEquals((int) integerList.get(0), 1);
357         assertEquals(stringList.get(0), "abc");
358         assertEquals(testClasses.get(0).getF(), 1.5f, 0);
359         assertEquals(boolResult, true);
360         assertEquals(innerWrapperResult, innerWrapper);
361         assertEquals(iObjectResult, iObject);
362         assertNotNull(mapResult.get("abc"));
363         assertSame(iObjects[0], message);
364         assertSame(iObjects[1], context);
365         assertSame(iObjects[2], response);
366 
367         inst.setIntValue(2);
368         inst.setStringValue("cba");
369         TestClass testClassForSetter = new TestClass();
370         inst.setTestClassValue(testClassForSetter);
371         List<Integer> integerListForSetter = new ArrayList<Integer>(){{add(2);}};
372         inst.setListOfInt(integerListForSetter);
373         List<String> stringListForSetter = new ArrayList<String>(){{add("cba");}};
374         inst.setListOfString(stringListForSetter);
375         List<TestClass> testClassListForSetter = new ArrayList<TestClass>(){{add(testClassForSetter);}};
376         inst.setListOfTestClasses(testClassListForSetter);
377         inst.setBoolValue(false);
378         IInnerWrapper innerWrapperForSetter = mock(IInnerWrapper.class);
379         inst.setWrappedIObject(innerWrapperForSetter);
380         IObject iObjectForSetter = mock(IObject.class);
381         inst.setIObject(iObjectForSetter);
382         inst.setStringIInnerMap(new HashMap<String, IInnerWrapper>(){{put("cba", innerWrapperForSetter);}});
383 
384         verify(response, times(1)).setValue(new FieldName("IntValue"), 2);
385         verify(response, times(1)).setValue(new FieldName("StringValue"), "cba");
386         verify(response, times(1)).setValue(new FieldName("TestClassValue"), testClassForSetter);
387         verify(response, times(1)).setValue(new FieldName("ListOfInt"), integerListForSetter);
388         verify(response, times(1)).setValue(new FieldName("ListOfString"), stringListForSetter);
389         verify(response, times(1)).setValue(new FieldName("ListOfTestClasses"), testClassListForSetter);
390         verify(response, times(1)).setValue(new FieldName("BoolValue"), false);
391         verify(response, times(1)).setValue(new FieldName("WrappedIObject"), innerWrapperForSetter);
392         verify(response, times(1)).setValue(new FieldName("IObject"), iObjectForSetter);
393         verify(context, times(1)).setValue(new FieldName("StringIInnerMap"), new HashMap<String, IInnerWrapper>(){{put("cba", innerWrapperForSetter);}});
394     }
