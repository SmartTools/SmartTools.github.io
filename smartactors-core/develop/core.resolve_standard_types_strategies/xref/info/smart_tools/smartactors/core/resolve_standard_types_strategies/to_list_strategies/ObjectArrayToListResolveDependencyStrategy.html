4   import info.smart_tools.smartactors.core.iresolve_dependency_strategy.exception.ResolveDependencyStrategyException;
5   
6   import java.util.Arrays;
7   
8   /**
9    * Strategy for converting from array to list
10   */
11  public class ObjectArrayToListResolveDependencyStrategy implements IResolveDependencyStrategy {
12  
13      @Override
14      public <T> T resolve(final Object... args) throws ResolveDependencyStrategyException {
15          try {
16              return (T) Arrays.asList(args);
17          } catch (Exception e) {
18              throw new ResolveDependencyStrategyException("Can't create list from array.", e);
19          }
20      }
21  }
