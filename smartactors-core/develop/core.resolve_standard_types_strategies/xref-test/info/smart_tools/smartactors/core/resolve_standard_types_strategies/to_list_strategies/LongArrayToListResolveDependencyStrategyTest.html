25          long[] array = new long[] {1, 2L, 5, Long.MIN_VALUE};

28          assertEquals(result.get(0), new Long(1));
29          assertEquals(result.get(1), new Long(2));
30          assertEquals(result.get(2), new Long(5));
31          assertEquals(result.get(3), new Long(Long.MIN_VALUE));
32      }
