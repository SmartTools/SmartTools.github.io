16      @Test
17      public void checkExpansibleURLClassLoaderCreation() {
18          ExpansibleURLClassLoader cl1 = new ExpansibleURLClassLoader(new URL[]{});
19          assertNotNull(cl1);
20          ExpansibleURLClassLoader cl2 = new ExpansibleURLClassLoader(new URL[]{}, ClassLoader.getSystemClassLoader());
21          assertNotNull(cl2);
22          assertSame(cl2.getParent(), ClassLoader.getSystemClassLoader());
23      }
24  
25      @Test
26      public void checkAdditionNewURL()
27              throws Exception {
28          URL url1 = new URL("http", "host", 9000, "filename1");
29          URL url2 =  new URL("http", "host", 9000, "filename2");
30          URL url3 =  new URL("http", "host", 9000, "filename1");
31          ExpansibleURLClassLoader cl = new ExpansibleURLClassLoader(new URL[]{url1});
32          assertNotNull(cl);
33          cl.addUrl(url2);
34          cl.addUrl(url3);
35          URL[] result = cl.getURLs();
36          assertSame(result[0], url1);
37          assertSame(result[1], url2);
38          assertEquals(result.length, 2);
39      }
